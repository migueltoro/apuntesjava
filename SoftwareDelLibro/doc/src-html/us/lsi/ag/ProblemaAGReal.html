001package us.lsi.ag;
002
003import java.util.List;
004
005import us.lsi.common.Par;
006
007/**
008 * @author Miguel Toro
009 *
010 * @param <S> Tipo de la solución
011 */
012public interface ProblemaAGReal<S> {
013        
014        /**
015         * @return Número de variables
016         */
017        Integer getNumeroDeVariables(); 
018        
019        /**
020         * @return Límites para cada variable
021         */
022        List<Par<Double,Double>> getLimites();
023        
024        /**
025         * @param ls Lista de valores de las variables
026         * @pre ls.size() == getNumeroDeVariables()
027         * @return Fitness del cromosoma
028         */
029        Double fitnessFunction(List<Double> ls);
030        
031        /**
032         * @param chromosome Es un cromosoma
033         * @return La solución definida por el cromosoma
034         */
035        S getSolucion(Cromosoma<Double> chromosome);
036        
037}
