001package us.lsi.ag;
002
003import java.util.List;
004
005import com.google.common.base.Preconditions;
006
007/**
008 * @author Miguel Toro
009 *
010 * @param <S> Tipo de la solución
011 */
012public interface ProblemaAGBag<S> {     
013        
014        
015        /**
016         * @return Numero de objetos distintos
017         */
018        Integer getNumeroDeObjetos();
019        
020        /**
021         * @pre <code> 0 &le; index &lt; getObjetos().size() </code>
022         * @param index Indice en la lista de objetos disponibles
023         * @return La multiplicidad máxima del objeto de índice <code> index </code>
024         */
025        
026        default Integer getMultiplicidadMaxima(int index){
027                Preconditions.checkElementIndex(index, this.getNumeroDeObjetos());
028                return 1;
029        }
030        
031        
032        /**
033         * @param ls Lista de índices del cromosoma
034         * @pre ls.size() == getDimension()
035         * @return Fitness del cromosoma
036         */
037        Double fitnessFunction(List<Integer> ls);
038        
039        /**
040         * @param chromosome Es un cromosoma
041         * @return La solución definida por el cromosoma
042         */
043        S getSolucion(Cromosoma<Integer> chromosome);
044}
