001package us.lsi.ag.reinas;
002
003import java.util.List;
004import java.util.Set;
005
006import com.google.common.collect.Lists;
007import com.google.common.collect.Sets;
008
009import us.lsi.ag.IndexChromosome;
010import us.lsi.ag.IndexProblemAG;
011import us.lsi.bt.reinas.Reina;
012
013public class ProblemaReinasAG implements IndexProblemAG<List<Reina>> {
014
015public static int numeroDeReinas = 8;
016        
017        public static ProblemaReinasAG create() {
018                return new ProblemaReinasAG();
019        }
020
021        private ProblemaReinasAG() {
022        }
023
024        @Override
025        public List<Reina> getSolucion(IndexChromosome chromosome) {
026                List<Integer> ls = chromosome.decode();
027                List<Reina> r = Lists.newArrayList();
028                for (int i = 0; i < ls.size(); i++) {
029                        r.add(Reina.create(i, ls.get(i)));
030                }
031                return r;
032        }
033
034        @Override
035        public Double fitnessFunction(IndexChromosome chromosome) {
036                List<Integer> ls = chromosome.decode();
037                Set<Integer> dp = Sets.newHashSet();
038                Set<Integer> ds = Sets.newHashSet();
039                for (int i = 0; i < ls.size(); i++) {
040                        dp.add(ls.get(i)-i);
041                        ds.add(ls.get(i)+i);
042                }
043                Double r = 2.*ProblemaReinasAG.numeroDeReinas-dp.size()-ds.size();
044                return -2000*r*r;
045        }
046
047        @Override
048        public Integer getObjectsNumber() {
049                return numeroDeReinas;
050        }       
051        
052
053}
