001package us.lsi.ag;
002
003import java.util.Comparator;
004import java.util.List;
005import java.util.function.Predicate;
006
007import org.apache.commons.math3.genetics.Chromosome;
008import org.apache.commons.math3.genetics.FixedGenerationCount;
009import org.apache.commons.math3.genetics.Population;
010import org.apache.commons.math3.genetics.StoppingCondition;
011
012import com.google.common.collect.Lists;
013
014import us.lsi.stream.Stream2;
015
016/**
017 * 
018 * <p> Acumula una lista, en la propiedad 
019 * <code> AlgoritmoAG.bestChromosomes </code> que guarda los mejores <code> numBestChromosomes </code> 
020 * distintos de cada generación.</p>
021 * 
022 * 
023 * <p> Implementa una condición de parada que se cumple cuando se encuentran <code> numBestChromosomes </code> cromosomas
024 * que cumplan un  <code> predicate </code> o el número de generaciones supera
025 * <code> AlgoritmoAG.NUM_GENERATIONS</code>  </p>
026 * 
027 * <p> La propiedad <code> predicate </code> debe ser inicializada desde el problema concreto a resolver. La propiedad numeroDeGeneraciones
028 * guarda el número de generaciones </p>
029 * 
030 * @author Miguel Toro
031 *
032 */
033public class SolutionsNumber implements StoppingCondition {
034
035        public static Integer numeroDeGeneraciones = 0;
036        public static Predicate<Chromosome> predicate =null;
037        private Integer ns;
038        private FixedGenerationCount fixed;
039        public SolutionsNumber(Integer numBestChromosomes, Integer NUM_GENERATIONS) {
040                super();
041                this.ns = numBestChromosomes;
042                AlgoritmoAG.bestChromosomes = Lists.newArrayList();
043                this.fixed = new FixedGenerationCount(NUM_GENERATIONS);
044                predicate = (Chromosome x) -> x.fitness() >= AlgoritmoAG.FITNESS; 
045        }
046
047        @Override
048        public boolean isSatisfied(Population population) {
049                List<Chromosome> ls = Stream2
050                                .<Chromosome> fromUnmodifiableIterable(population)
051                                .sorted(Comparator.<Chromosome> reverseOrder())
052                                .distinct()
053                                .limit(ns)
054                                .toList();
055                AlgoritmoAG.bestChromosomes.addAll(ls);
056                boolean ret = fixed.isSatisfied(population);
057                ret = ret || ls.stream().allMatch((Chromosome x) -> predicate.test(x));
058                numeroDeGeneraciones++;
059                return ret;
060        }
061
062}
