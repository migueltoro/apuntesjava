001package us.lsi.ag;
002
003import org.apache.commons.math3.genetics.BinaryMutation;
004import org.apache.commons.math3.genetics.Chromosome;
005
006
007public class AlgoritmoAGReal extends AlgoritmoAG<Integer> {
008        
009        public AlgoritmoAGReal(ProblemaAGReal<?> problema) {
010                super();
011                super.mutationPolicy = new BinaryMutation();
012                AlgoritmoAG.DIMENSION = problema.getNumeroDeVariables()*RealBinaryChromosome.numeroDeBits;
013        }
014        
015        @Override
016        public Chromosome getInitialChromosome() {
017                return RealBinaryChromosome.getInitialChromosome();
018        }
019        
020}
