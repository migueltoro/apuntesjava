003
004import java.util.List;
005import java.util.stream.Collectors;
006import java.util.stream.IntStream;
007
008import   org.apache.commons.math3.genetics.*;
009
010import us.lsi.common.Lists2;
011
012
013/**
014 * <p> Algoritmo específico para algoritmos genéticos con codificación real </p>
015 * <p>Es adecaudo para resolver problemas que se representan por permutacionanes </p> 
016 * <p>Los cromosomas que implementemos deben heredar de la clase 
017 * <a href="http://commons.apache.org/proper/commons-math/apidocs/org/apache/commons/math3/genetics/RandomKey.html" target="_blank"> RandomKey </a>
018 * </p>
019 * 
020 * 
021 * @author Miguel Toro
022 *
023 * 
024 */
025public class AlgoritmoAGRandomKey extends AlgoritmoAG<Double> {
026        
027        private static List<Integer> getNormalSequence(ProblemaAGBag<?> problema) {
028                return IntStream.range(0,problema.getNumeroDeObjetos())
029                                .boxed()
030                                .map(x->Lists2.nCopias(problema.getMultiplicidadMaxima(x),x).stream())
031                                .flatMap(x->x)
032                                .collect(Collectors.toList());
033        }
034        
035        public AlgoritmoAGRandomKey(ProblemaAGBag<?> problema) {
036                super();
037                super.mutationPolicy = new RandomKeyMutation();
038                AlgoritmoAG.DIMENSION = IntStream.range(0,problema.getNumeroDeObjetos()).map(x->problema.getMultiplicidadMaxima(x)).sum();
039                BagRandomKeyChromosome.normalSequence = getNormalSequence(problema);
040                
041        }
042        
043        @Override
044        public Chromosome getInitialChromosome() {
045                return BagRandomKeyChromosome.getInitialChromosome();
046        }
047        
048        
049        
050}
