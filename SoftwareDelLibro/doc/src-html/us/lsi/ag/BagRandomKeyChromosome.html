001package us.lsi.ag;
002
003import java.util.List;
004import java.util.function.Function;
005
006import org.apache.commons.math3.genetics.AbstractListChromosome;
007import org.apache.commons.math3.genetics.InvalidRepresentationException;
008import org.apache.commons.math3.genetics.RandomKey;
009
010import com.google.common.base.Preconditions;
011
012/**
013 * @author Miguel Toro
014 * 
015 * <p> Una implementación del tipo Cromosoma&lt;Integer&gt;. Toma como información la definición de un problema que implementa el interfaz ProblemaAGBag.
016 * A partir de esa información construye una secuencia normal. Asumimos que el número de objetos es n y el tamaño de la secuencia normal r. 
017 *  
018 * La lista decodificada está formada por una lista de  tamaño r, cuyos valores son 
019 * índices en el rango [0,n-1], y cada índice i se  repite un número de veces dado por la multiplicidad máxima del objeto i
020 * definida en el problema.
021 * 
022 * La implementación usa un cromosoma de clave aleatoria de tamaño r.
023 * Es un cromosoma adecuado para codificar problemas de permutaciones </p>
024 *
025 */
026public class BagRandomKeyChromosome extends RandomKey<Integer> implements Cromosoma<Integer> {
027
028        public static Function<List<Integer>,Double> fitnessFunction;
029        public static List<Integer> normalSequence;
030        
031        public BagRandomKeyChromosome(List<Double> representation)
032                        throws InvalidRepresentationException {
033                super(representation);
034                this.ft = this.calculateFt();
035        }
036
037        public BagRandomKeyChromosome(Double[] representation)
038                        throws InvalidRepresentationException {
039                super(representation);
040                this.ft = this.calculateFt();
041        }
042
043        @Override
044        public AbstractListChromosome<Double> newFixedLengthChromosome(List<Double> ls) {
045                return new BagRandomKeyChromosome(ls);
046                
047        }
048
049        @Override
050        public List<Integer> decode() {
051                Preconditions.checkArgument(BagRandomKeyChromosome.normalSequence!=null);
052                return this.decode(BagRandomKeyChromosome.normalSequence);
053        }
054        
055        
056        public static BagRandomKeyChromosome getInitialChromosome() {
057                List<Double> ls = RandomKey.randomPermutation(AlgoritmoAG.DIMENSION);
058                return new BagRandomKeyChromosome(ls);
059        }
060
061        
062        @Override
063        public double fitness() {
064                return ft;
065        }
066        
067        private double ft;
068        
069        private double calculateFt(){
070                return BagRandomKeyChromosome.fitnessFunction.apply(this.decode());
071        }       
072}
