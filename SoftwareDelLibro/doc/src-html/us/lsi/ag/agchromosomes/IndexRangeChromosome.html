001package us.lsi.ag.agchromosomes;
002
003import java.util.ArrayList;
004import java.util.List;
005
006import org.apache.commons.math3.genetics.AbstractListChromosome;
007import org.apache.commons.math3.genetics.BinaryChromosome;
008import org.apache.commons.math3.genetics.Chromosome;
009import org.apache.commons.math3.genetics.InvalidRepresentationException;
010
011import us.lsi.ag.ProblemaAG;
012import us.lsi.ag.ProblemaAGIndex;
013import us.lsi.common.Lists2;
014import us.lsi.math.Math2;
015
016/**
017 * @author Miguel Toro
018 * 
019 * 
020 * <p> Una implementación del tipo Cromosoma&lt;Integer&gt;. Toma como información la definición de un problema que implementa el interfaz ProblemaAGIndex.
021 * Asumimos que el número de objetos es n. Usa una  secuencia normal implícita de tamaño n de la forma (0,1,2, ..., n-1). </p>
022 * 
023 * <p> La lista decodificada está formada por una lista de  tamaño n cuyos elementos para cada i son 
024 *  valores en en rango [0,m(i)], siendo m(i) la multiplicidad máxima para i. </p>
025 * 
026 * <p> La implementación usa un cromosoma binario del tamaño n*nbits. 
027 * Siendo nbits el número de bits usados para representar cada uno de los enteros. </p>
028 * 
029 * <p> Es un cromosoma adecuado para codificar problemas de subconjuntos de multiconjuntos</p>
030 *
031 */
032public class IndexRangeChromosome extends BinaryChromosome implements IndexChromosome {
033        
034        /**
035         * Número de bits usado para representar un entero. El rango de enteros que podemos obtener dependerá de este número de bits.
036         */
037        public static Integer numeroDeBits = 5;
038        
039        public static ProblemaAGIndex<?> problema;
040        
041        /**
042         * Dimensión del cromosoma igual a numeroDeBits*getObjectsNumber()
043         */
044        
045        public static int DIMENSION;
046        
047        public static void iniValues(ProblemaAG problema){
048                IndexRangeChromosome.problema = (ProblemaAGIndex<?>) problema; 
049                IndexRangeChromosome.DIMENSION = IndexRangeChromosome.numeroDeBits*IndexRangeChromosome.problema.getObjectsNumber();
050        }
051        
052        private static Integer pow = Math2.pow(2., numeroDeBits).intValue();
053        
054        public IndexRangeChromosome(Integer[] representation) throws InvalidRepresentationException {
055                super(representation);
056                this.ft = this.calculateFt();
057        }
058
059        public IndexRangeChromosome(List<Integer> representation) throws InvalidRepresentationException {
060                super(representation);
061                this.ft = this.calculateFt();
062        }
063
064        @Override
065        public AbstractListChromosome<Integer> newFixedLengthChromosome(List<Integer> ls) {
066                return new IndexRangeChromosome(ls);
067        }
068        
069        public List<Integer> decode() {
070                List<Integer> ls = super.getRepresentation();
071                List<Integer> r = new ArrayList<Integer>();
072                int index1 = 0;
073                for(int i = 0; i < getObjectsNumber(); i++){                    
074                        int index2 = index1+numeroDeBits;
075                        Integer e = Lists2.decode(ls.subList(index1, index2));
076                        Integer d = Math2.escalaIncluded(e, pow, getMax(i));
077                        r.add(d);
078                        index1 = index1+numeroDeBits;;
079                }
080                return r;
081        }
082
083        public List<Integer> getRepresentation(){
084                return super.getRepresentation();
085        }
086        
087        public static IndexRangeChromosome getInitialChromosome() {
088                List<Integer> ls = BinaryChromosome.randomBinaryRepresentation(IndexRangeChromosome.DIMENSION);
089                return new IndexRangeChromosome(ls);
090        }
091
092        @Override
093        public double fitness() {
094                return ft;
095        }
096        
097        private double ft;
098        
099        private double calculateFt(){
100                return IndexRangeChromosome.problema.fitnessFunction(this);
101        }
102
103        public Integer getObjectsNumber() {
104                return IndexRangeChromosome.problema.getObjectsNumber();
105        }
106
107        public Integer getMax(int i) {
108                return IndexRangeChromosome.problema.getMax(i);
109        }
110
111        @Override
112        public ProblemaAGIndex<?> getProblem() {
113                return IndexRangeChromosome.problema;
114        }
115
116        @Override
117        public Chromosome asChromosome() {
118                return this;
119        }
120        
121        
122        
123}
