001package us.lsi.ag.agchromosomes;
002
003import java.util.ArrayList;
004import java.util.List;
005
006import org.apache.commons.math3.genetics.AbstractListChromosome;
007import org.apache.commons.math3.genetics.BinaryChromosome;
008import org.apache.commons.math3.genetics.Chromosome;
009import org.apache.commons.math3.genetics.InvalidRepresentationException;
010
011import com.google.common.base.Preconditions;
012
013import us.lsi.ag.ProblemaAG;
014import us.lsi.ag.ProblemaAGReal;
015import us.lsi.common.Lists2;
016
017/**
018 * @author Miguel Toro
019 * 
020 * 
021 * <p> Una implementación del tipo Cromosoma&lt;Double&gt;. Toma como información la definición de un problema que implementa el interfaz ProblemaAGReal.
022 * A partir de esa información construye una secuencia normal. Asumimos que el número de objetos es n y el tamaño de la secuencia normal r. </p>
023 *  
024 * <p> La lista decodificada está formada por una lista de  tamaño nv,  cuyos elementos son valores en reales en el rango [li(i),ls(i)].</p>
025 * 
026 * 
027 * 
028 * <p> La implementación usa un cromosoma binario  de tamaño nv*nbits.
029 * Es un cromosoma adecuado para codificar problemas de funciones  reales en un espacio de n dimensiones </p>
030 *
031 */
032public class RealListChromosome extends BinaryChromosome implements IRealChromosome {
033
034        
035        public static Integer numeroDeBits = 10;
036        private static Double fact = null;
037        public static ProblemaAGReal<?> problema;
038        
039        /**
040         * Dimensión del cromosoma
041         */
042        
043        public static int DIMENSION;
044        
045        public static void iniValues(ProblemaAG problema){
046                RealListChromosome.problema = (ProblemaAGReal<?>) problema;
047                RealListChromosome.DIMENSION = RealListChromosome.problema.getNumeroDeVariables()*RealListChromosome.numeroDeBits;
048        }
049        
050        public RealListChromosome(Integer[] representation) throws InvalidRepresentationException {
051                super(representation);
052                this.ft = this.calculateFt();
053        }
054
055        public RealListChromosome(List<Integer> representation) throws InvalidRepresentationException {
056                super(representation);
057                this.ft = this.calculateFt();
058        }
059        
060        private Double getFact() {
061                if(fact==null){
062                        fact = (double) (Math.pow(2,numeroDeBits)-1);
063                }
064                return fact;
065        }
066
067        @Override
068        public AbstractListChromosome<Integer> newFixedLengthChromosome(List<Integer> ls) {
069                return new RealListChromosome(ls);
070        }
071        @Override
072        public List<Double> decode() {
073                List<Integer> ls = super.getRepresentation();
074                List<Double> r = new ArrayList<Double>();
075                for(int i = 0; i< this.getNum(); i++){
076                        int index1 = i*numeroDeBits;
077                        int index2 = index1+numeroDeBits;
078                        Preconditions.checkElementIndex(index1, ls.size());
079                        Preconditions.checkPositionIndex(index2, ls.size());
080                        Double e = (double) Lists2.decode(ls.subList(index1, index2));
081                        Double x = this.getInf(i);
082                        Double y = this.getSup(i);
083                        Double d = x+(y-x)*e/this.getFact();
084                        r.add(d);
085                }
086                return r;
087        }
088        @Override
089        public List<Integer> getRepresentation(){
090                return super.getRepresentation();
091        }
092        
093        public static RealListChromosome getInitialChromosome() {
094                List<Integer> ls = BinaryChromosome.randomBinaryRepresentation(RealListChromosome.DIMENSION);
095                return new RealListChromosome(ls);
096        }
097
098        @Override
099        public double fitness() {
100                return ft;
101        }
102        
103        private double ft;
104        
105        private double calculateFt(){
106                return RealListChromosome.problema.fitnessFunction(this);
107        }
108        
109        @SuppressWarnings("unchecked")
110        public static IChromosome<List<Double>> asCromosoma(Chromosome cr){
111                return (IChromosome<List<Double>>) cr;
112        }
113
114        /* (non-Javadoc)
115         * @see us.lsi.ag.IRealChromosome#getNum()
116         */
117        @Override
118        public Integer getNum() {
119                return RealListChromosome.problema.getNumeroDeVariables();
120        }
121
122        /* (non-Javadoc)
123         * @see us.lsi.ag.IRealChromosome#getSup(int)
124         */
125        @Override
126        public Double getSup(int i) {
127                return RealListChromosome.problema.getLimites().get(i).p2;
128        }
129
130        /* (non-Javadoc)
131         * @see us.lsi.ag.IRealChromosome#getInf(int)
132         */
133        @Override
134        public Double getInf(int i) {
135                return RealListChromosome.problema.getLimites().get(i).p1;
136        }
137
138        @Override
139        public ProblemaAGReal<?> getProblema() {
140                return RealListChromosome.problema;
141        }
142        
143        @Override
144        public Chromosome asChromosome() {
145                return this;
146        }
147}
