033         * Los diferentes tipos de cromosmomas implementados
034         *
035         */
036        public enum ChromosomeType {Binary,ListInteger,IndexSubList,IndexRange,IndexPermutation,IndexPermutationSubList,Real,Expression}
037        
038        public static ChromosomeType tipo;
039
040        /**
041         * @param tipo El tipo de cromosoma
042         * @return Un cromosoma aleatorio del tipo indicado
043         */
044        public static IChromosome<?> randomChromosome(ChromosomeType tipo){
045                IChromosome<?> chromosome = null;
046                switch(tipo){
047                case Binary: chromosome = BinaryChromosome2.getInitialChromosome(); break;
048                case IndexSubList: chromosome = IndexSubListChromosome.getInitialChromosome(); break;
049                case ListInteger: chromosome = ListIntegerChromosome.getInitialChromosome(); break;
050                case IndexRange: chromosome = IndexRangeChromosome.getInitialChromosome(); break;
051                case IndexPermutation: chromosome = IndexPermutationChromosome.getInitialChromosome(); break;
052                case IndexPermutationSubList: chromosome = IndexPermutationSubListChromosome.getInitialChromosome(); break;
053                case Real: chromosome = RealChromosome.getInitialChromosome(); break;
054                case Expression: chromosome = ExpressionChromosome.getInitialChromosome(); break;
055                }
056                return chromosome;
057        }
058        
059        /**
060         * <p>Distintos tipo de operadores de cruce </p>
061         *
062         * <ul>
063         * <li> <a href="http://commons.apache.org/proper/commons-math/apidocs/org/apache/commons/math3/genetics/OnePointCrossover.html" target="_blank"> OnePointCrossover </a>
064         * <li> <a href="http://commons.apache.org/proper/commons-math/apidocs/org/apache/commons/math3/genetics/NPointCrossover.html" target="_blank"> NPointCrossover </a>
065         * <li> <a href="http://commons.apache.org/proper/commons-math/apidocs/org/apache/commons/math3/genetics/CycleCrossover.html" target="_blank"> CycleCrossover </a>
066         * <li> <a href="http://commons.apache.org/proper/commons-math/apidocs/org/apache/commons/math3/genetics/OrderedCrossover.html" target="_blank"> OrderedCrossover </a>
067         * <li> <a href="http://commons.apache.org/proper/commons-math/apidocs/org/apache/commons/math3/genetics/UniformCrossover.html" target="_blank"> UniformCrossover </a>
068         * </ul> 
069         * 
070         */
071        
072        public enum CrossoverType{Cycle,NPoint,OnePoint,Ordered,Uniform};
073        
074        /**
075         * Tipo del operador de cruce
076         */
077        public static CrossoverType crossoverType = CrossoverType.OnePoint;
078        
079        /**
080         * Número de puntos usados en la partición si se usa un operador de cruce de tipo NPointCrossover
081         */
082        public static int NPOINTCROSSOVER = 3;
083        /**
084         * La ratio si se usa el operador de cruce de tipo UniformCrossover
085         */
086        public static double RATIO_UNIFORMCROSSOVER = 0.7;
087        
088        /**
089         * @param tipo El tipo del cromosoma
090         * @return Un operador de cruce adecuado para un cromosma del tipo indicado
091         */
092        public static CrossoverPolicy getCrossoverPolicy(ChromosomeType tipo){
093                CrossoverPolicy crossOverPolicyBin = null;      
094                switch(crossoverType){
095                case Cycle: crossOverPolicyBin = new CycleCrossover<Integer>();break;
096                case NPoint: crossOverPolicyBin = new NPointCrossover<Integer>(NPOINTCROSSOVER);break;
097                case OnePoint: crossOverPolicyBin = new OnePointCrossover<Integer>();break;
098                case Ordered: crossOverPolicyBin = new OrderedCrossover<Integer>(); break;
099                case Uniform: crossOverPolicyBin = new UniformCrossover<Integer>(RATIO_UNIFORMCROSSOVER); break;
100                }
101                CrossoverPolicy crossOverPolicyKey = null;      
102                switch(crossoverType){
103                case Cycle: crossOverPolicyKey = new CycleCrossover<Double>();break;
104                case NPoint: crossOverPolicyKey = new NPointCrossover<Double>(NPOINTCROSSOVER);break;
105                case OnePoint: crossOverPolicyKey = new OnePointCrossover<Double>();break;
106                case Ordered: crossOverPolicyKey = new OrderedCrossover<Double>(); break;
107                case Uniform: crossOverPolicyKey = new UniformCrossover<Double>(RATIO_UNIFORMCROSSOVER); break;
108                }
109                CrossoverPolicy crossOverPolicy = null; 
110                switch(tipo){
111                case Binary: crossOverPolicy = crossOverPolicyBin; break;
112                case IndexSubList: crossOverPolicy = crossOverPolicyBin; break;
113                case ListInteger: crossOverPolicy = crossOverPolicyBin; break;
114                case IndexRange: crossOverPolicy = crossOverPolicyBin; break;
115                case IndexPermutation: crossOverPolicy = crossOverPolicyKey; break;
116                case IndexPermutationSubList: crossOverPolicy = new SubListCrossoverPolicy(crossOverPolicyBin,crossOverPolicyKey); break;
117                case Real: crossOverPolicy = crossOverPolicyBin; break;
118                case Expression: crossOverPolicy = crossOverPolicyBin; break;
119                }
120                Preconditions.checkState(crossOverPolicy!=null);
121                return crossOverPolicy;
122        }
123        
124        /**
125         * @param tipo El tipo del cromosoma
126         * @param problema El problema a resolver
127         * @return Un operador de mutación adecuado para un cromosoma del tipo indicado
128         */
129        public static MutationPolicy getMutationPolicy(ChromosomeType tipo, ProblemaAG problema){
130                MutationPolicy mutationPolicy = null;
131                switch(tipo){
132                case Binary:  mutationPolicy = new BinaryMutation()     ; break;
133                case IndexSubList: mutationPolicy = new BinaryMutation(); break;
134                case ListInteger: mutationPolicy = ((ProblemaAGListInteger<?>)problema).getMutationPolicy(); break;
135                case IndexRange: mutationPolicy = new BinaryMutation(); ; break;
136                case IndexPermutation: mutationPolicy = new RandomKeyMutation(); break;
137                case IndexPermutationSubList: mutationPolicy = new SubListMutationPolicy(); break;
138                case Real: mutationPolicy = new BinaryMutation();       ; break;
139                case Expression: mutationPolicy = new BinaryMutation(); ; break;
140                }
141                Preconditions.checkState(mutationPolicy!=null);
142                return mutationPolicy;
143        }
144        
145        /**
146         * <p> Para aplicar los operadores de mutación se escogen dos cromosomas. 
147         * La técnica implementada para escoger cada uno de los dos cromosomas se denomina selección por torneo. 
148         * Se trata de organizar dos torneos. 
149         * En cada uno se elige el mejor cromosoma de entrre <code> TOURNAMENT_ARITY </code> cromosomas de la población seleccionados al azar. 
150         * Si el tamaño de <code> TOURNAMENT_ARITY </code> es más grande, los cromosomas
151         *  débiles tienen menor probabilidad de ser seleccionados.</p>
152         * 
153         * <p> Número de participantes en el torneo para elegir los cromosomas que participarán en el cruce </p>
154         * <p> Un valor típico es 2 </p>
155         */
156        
157        public static int TOURNAMENT_ARITY = 2;
158        
159        /**
160         * @return Un operador que implementa la política de selección escogida
161         */
162        public static SelectionPolicy getSelectionPolicy(){     
163                SelectionPolicy selectionPolicy = new TournamentSelection(TOURNAMENT_ARITY);
164                Preconditions.checkState(selectionPolicy!=null);
165                return selectionPolicy;
166        }

168        /**
169         * @param tipo Tipo de cromosoma
170         * @param problema El problema a resolver 
171         * @pos El método inicializa los parámetros relevantes de la clase que implementa el tipo indicado de cromosoma
172         */
173        public static void iniValues(ChromosomeType tipo, ProblemaAG problema){
174                switch(tipo){
175                case Binary: BinaryChromosome2.iniValues(problema);break;
176                case IndexSubList: IndexSubListChromosome.iniValues(problema);break;
177                case ListInteger: ListIntegerChromosome.iniValues(problema);break;
178                case IndexRange: IndexRangeChromosome.iniValues(problema); break;
179                case IndexPermutation: IndexPermutationChromosome.iniValues(problema);break;
180                case IndexPermutationSubList: IndexPermutationSubListChromosome.iniValues(problema);break;
181                case Real: RealChromosome.iniValues(problema);break;
182                case Expression: ExpressionChromosome.iniValues(problema);break;
183                }
184        }
