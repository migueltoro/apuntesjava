001package us.lsi.geometria;
002
003
004public class ParDePuntos implements Comparable<ParDePuntos> {
005
006        public static ParDePuntos create(Punto2D p1, Punto2D p2) {
007                return new ParDePuntos(p1, p2);
008        }
009
010        private Punto2D p1;
011        private Punto2D p2;
012        
013        private ParDePuntos(Punto2D p1, Punto2D p2) {
014                super();
015                this.p1 = p1.compareTo(p2)>=0?p1:p2;
016                this.p2 = p1.compareTo(p2)>=0?p2:p1;;
017        }
018
019        public Punto2D getP1() {
020                return p1;
021        }
022
023        public Punto2D getP2() {
024                return p2;
025        }
026        
027        public Double getDistancia(){
028                return p1.getDistanciaA(p2);
029        }
030
031        @Override
032        public int compareTo(ParDePuntos a) {
033                // TODO Auto-generated method stub
034                int r = this.getDistancia().compareTo(a.getDistancia());
035                if(r==0){
036                        r = this.getP1().compareTo(a.getP1());
037                }
038                if(r==0){
039                        r = this.getP2().compareTo(a.getP2());
040                }
041                return r;
042        }
043        
044        
045        @Override
046        public int hashCode() {
047                final int prime = 31;
048                int result = 1;
049                result = prime * result + ((p1 == null) ? 0 : p1.hashCode());
050                result = prime * result + ((p2 == null) ? 0 : p2.hashCode());
051                return result;
052        }
053
054        @Override
055        public boolean equals(Object obj) {
056                if (this == obj)
057                        return true;
058                if (obj == null)
059                        return false;
060                if (getClass() != obj.getClass())
061                        return false;
062                ParDePuntos other = (ParDePuntos) obj;
063                if (p1 == null) {
064                        if (other.p1 != null)
065                                return false;
066                } else if (!p1.equals(other.p1))
067                        return false;
068                if (p2 == null) {
069                        if (other.p2 != null)
070                                return false;
071                } else if (!p2.equals(other.p2))
072                        return false;
073                return true;
074        }
075
076        @Override
077        public String toString() {
078                String st1 = p1.compareTo(p2)<=0? p1.toString() : p2.toString();
079                String st2 = p1.compareTo(p2)>0? p1.toString() : p2.toString();         
080                return "[" + st1 + "," + st2 + ","+getDistancia()+"]";
081        }
082        
083        
084}
