018        public static <R> Exp<R> scan(String s) {
019                Tokenizer tk = Tokenizer.create(s, Operator.functions, Operator.reservedWords);
020                Map<String, Exp<?>> vars = new HashMap<>();
021                @SuppressWarnings("unchecked")
022                Exp<R> r = (Exp<R>) ExpParser.scanExp(tk, vars);
023                return r;
024        }

026        @SuppressWarnings("unchecked")
027        public static <R> Exp<R> scan(String s, Map<String, Exp<?>> vars) {
028                Tokenizer tk = Tokenizer.create(s, Operator.functions, Operator.reservedWords);
029                Map<String, Exp<?>> nVars = new HashMap<>(vars);
030            vars.values().stream().map(e->e.getVars()).forEach(m->nVars.putAll(m));             
031            Exp<R> r = (Exp<R>) ExpParser.scanExp(tk, nVars);
032                return r;
033        }

035        protected static Exp<?> scanExp(Tokenizer tk, Map<String,Exp<?>> vars){
036                Exp<?> r = scanParticleMultiply(tk,vars);
037                String s;
038                while (tk.hasMoreTokens() && 
039                                tk.seeNextTokenType().equals(TokenType.Operator) &&
040                                Operator.arities.get(tk.seeNextToken()) == 2) {
041                        s = tk.matchTokenTypes(TokenType.Operator);
042                        Exp<?> exp = ExpParser.scanParticleMultiply(tk, vars);
043                        r = Operator.getBinary(s,r.getExpType(),exp.getExpType()).exp(r,exp);
044                }
045                return r;
046        }
047        
048        protected static Exp<?> scanParticleMultiply(Tokenizer tk, Map<String,Exp<?>> vars){
049                Exp<?> r = scanParticlePot(tk,vars);
050                String s;
051                while (tk.hasMoreTokens() && 
052                                tk.seeNextTokenType().equals(TokenType.Operator) &&
053                                Operator.arities.get(tk.seeNextToken()) == 4) {
054                        s = tk.matchTokenTypes(TokenType.Operator);
055                        Exp<?> exp = ExpParser.scanParticlePot(tk, vars);
056                        r = Operator.getBinary(s,r.getExpType(),exp.getExpType()).exp(r,exp);
057                }
058                return r;
059        }
060        
061        protected static Exp<?> scanParticlePot(Tokenizer tk, Map<String,Exp<?>> vars){
062                Exp<?> r = scanParticleBasic(tk,vars);
063                String s;
064                if (tk.hasMoreTokens() && 
065                                tk.seeNextTokenType().equals(TokenType.Operator) &&
066                                Operator.arities.get(tk.seeNextToken()) == 6) {
067                        s = tk.matchTokenTypes(TokenType.Operator);
068                        Exp<?> exp = ExpParser.scanParticleBasic(tk, vars);
069                        r = Operator.getBinary(s,r.getExpType(),exp.getExpType()).exp(r,exp);
070                }
071                return r;
072        }
073        protected static Exp<?> scanParticleBasic(Tokenizer tk, Map<String,Exp<?>> vars){
074                Exp<?> r = null;
075                String s;
076                TokenType tt = tk.seeNextTokenType();
077                switch (tt) {   
078                case Integer:
079                        r = ExpParser.scanConstantInteger(tk);
080                        break;
081                case Double:
082                        r = ExpParser.scanConstantDouble(tk);
083                        break; 
084                case VariableIdentifier:
085                        if (tk.seeNextToken().charAt(0) == ('@')) {
086                                r = ExpParser.scanPlaceHolder(tk, vars);
087                        } else {
088                                r = ExpParser.scanVariable(tk, vars);
089                        }
090                        break;
091                case Operator:
092                case FunctionIdentifier:
093                        s = tk.matchTokenTypes(TokenType.Operator,TokenType.FunctionIdentifier);                        
094                        tk.matchTokens("(");    
095                        r = scanParameters(s,tk,vars);
096                        tk.matchTokens(")");
097                        break;
098                case Separator: 
099                        tk.matchTokens("(");
100                        if(Lists2.newList("int","double").contains(tk.seeNextToken())){
101                                r = scanCastType(tk, vars);
102                        } else {                                
103                                r = scanExp(tk, vars);
104                                tk.matchTokens(")");
105                        }                       
106                        break;
107                default:
108                        Preconditions.checkState(false, 
109                                        String.format("No se esperaba %s en la posición \n   %s",
110                                                        tk.getToken(),tk.getSufix()));                  
111                }                       
112                return r;
113        }

115        private static Exp<?> scanCastType(Tokenizer tk, Map<String, Exp<?>> vars){
116                String  op = tk.matchTokens("int","double");
117                tk.matchTokens(")");
118                Exp<?> exp = scanExp(tk, vars);
119                String name = "("+op+")";
120                return Operator.getUnary(name, exp.getExpType()).exp(exp);
121        }

123        private static Exp<?> scanParameters(String name, Tokenizer tk,
124                        Map<String, Exp<?>> vars) {
125                List<Exp<?>> exps = new ArrayList<>();
126                Exp<?> exp = ExpParser.scanExp(tk, vars);
127                exps.add(exp);
128                while (tk.seeNextToken().equals(",")) {
129                        tk.matchTokens(",");
130                        exp = ExpParser.scanExp(tk, vars);
131                        exps.add(exp);
132                }
133                Exp<?> r = null;
134                int arity = exps.size();
135                boolean sameType = IntStream.range(0, exps.size() - 1).allMatch(
136                                i -> exps.get(i).getExpType()
137                                                .equals(exps.get(i + 1).getExpType()));
138                if (sameType) {
139                        switch (arity) {
140                        case 1:r = Operator.getUnary(name, exp.getExpType()).exp(exp);break;
141                        case 2:r = Operator.getBinary(name, exps.get(0).getExpType(),
142                                                exps.get(1).getExpType()).exp(exps.get(0), exps.get(1));break;
143                        case 3:r = Operator.getTernary(name, exps.get(0).getExpType(),
144                                                exps.get(1).getExpType(), exps.get(2).getExpType())
145                                                .exp(exps.get(0), exps.get(1), exps.get(2));break;
146                        default:
147                                r = Operator.getAccumulator(name, exp.getExpType()).exp(exps);
148                        }
149                } else {
150                        switch (arity) {
151                        case 1:
152                                r = Operator.getUnary(name, exp.getExpType()).exp(exp);
153                                break;
154                        case 2:
155                                r = Operator.getBinary(name, exps.get(0).getExpType(),
156                                                exps.get(1).getExpType()).exp(exps.get(0), exps.get(1));
157                                break;
158                        case 3:
159                                r = Operator.getTernary(name, exps.get(0).getExpType(),
160                                                exps.get(1).getExpType(), exps.get(2).getExpType())
161                                                .exp(exps.get(0), exps.get(1), exps.get(2));
162                                break;
163                        default:
164                                Preconditions.checkState(false, String.format(
165                                                "No hay disponibles operadores de aridad %d", arity));
166                        }
167                }
168                return r;
169
170        }

172        public static VariableExp<?> scanVariable(Tokenizer tk, Map<String,Exp<?>> vars){               
173                String s = tk.matchTokenTypes(TokenType.VariableIdentifier);
174                VariableExp<?> r=null;  
175                if(vars.containsKey(s)){
176                r = (VariableExp<?>) vars.get(s);
177                } else {
178                        ExpType expType = Operator.getVariableType(s);
179                        switch(expType){
180                        case Boolean: r = Exp.<Boolean>variable(s, ExpType.Boolean);break;
181                        case Integer: r = Exp.<Integer>variable(s, ExpType.Integer);break;
182                        case Double:  r = Exp.<Double>variable(s, ExpType.Double);                              
183                        }
184                        vars.put(s, r);
185                }
186                return r;
187        }
188
189        public static Exp<?> scanPlaceHolder(Tokenizer tk, Map<String,Exp<?>> vars){                                            
190                String s = tk.matchTokenTypes(TokenType.VariableIdentifier);
191                Exp<?> r=null;  
192                if(vars.containsKey(s)){
193                r = vars.get(s);
194                } else {
195                        ExpType expType = Operator.getVariableType(s.substring(1));
196                        switch(expType){
197                        case Boolean: r = Exp.<Boolean>placeHolder(s, ExpType.Boolean);break;
198                        case Integer: r = Exp.<Integer>placeHolder(s, ExpType.Integer);break;
199                        case Double:  r = Exp.<Double>placeHolder(s, ExpType.Double);                           
200                        }
201                        vars.put(s, r);
202                }
203                return r;
204        }

206        public static ConstantExp<Double> scanConstantDouble(Tokenizer tk){             
207                String s = tk.matchTokenTypes(TokenType.Double);
208                return Exp.constant(new Double(s),ExpType.Double);
209        }

211        public static ConstantExp<Integer> scanConstantInteger(Tokenizer tk){           
212                String s = tk.matchTokenTypes(TokenType.Integer);
213                return Exp.constant(new Integer(s),ExpType.Integer);
