001package us.lsi.bt;
002
003import java.util.List;
004
005
006/**
007 * <p> Tipo mutable del estado de un problema que se quiere resolver por la técnica de Bactracking. 
008 * Cada estado está asociado un problema, del conjunto de problemas considerado, más el camino, secuencia de alternativas, para llegar a él desde el problema inicial. </p>
009 * 
010 * <p>La documentación puede encontarse en el: <a href="../../../document/Tema15.pdf" target="_blank">Tema15</a></p>
011 *
012 * @author Miguel Toro
013 *
014 * @param <S> El tipo de la solución
015 * @param <A> El tipo de la alternativa
016 */
017public interface EstadoBT<S, A> {
018        /**
019         * @param a - alternativa escogida para cambiar el estado
020         */
021        void avanza(A a);
022        /**
023         * @param a - alternativa a eliminar para restaurar el estado anterior
024         */
025        void retrocede(A a);
026        /**
027         * @return Tamaño del problema al que está asociado el estado
028         */
029        int size();
030        /**
031         * @return Si el problema asociado al estado es final
032         */
033        boolean isFinal();
034        /**
035         * @return Las alternativas admisibles para el problema asociado
036         */
037        List<A> getAlternativas();
038        /**
039         * <p> Para poder calcular la solución del problema inicial en el estado final es muy recomendable que el estado disponga de una propiedad de tipo List&lt;A&gt; 
040         * que almacena se secuencia de alternativas tomadas. </p>.
041         * 
042         * @pre isFinal()
043         * @return Una solución del problema inicial y null si no hay solución para el problema inicial tras la secuencia de alternativas tomadas
044         * @throws IllegalStateException - Si se invoca el método en un estado no final
045         */
046        S getSolucion();
047        
048        /**
049         * @pos Si no es posible calcularla  se devolverá MAX_VALUE si el problema es de minimización y MIN_VALUE si es de maximización. 
050         * @return Valor de propiedad objetivo del problema inicial asumiendo que estamos en el problema final.    
051         */
052        Double getObjetivo();
053        
054        /**
055         * @pos 
056         * <ul>
057         * <li> Si el problema es de minimización el valor debe ser 
058         * una cota inferior del valor de la propiedad objetivo del problema inicial, 
059         * asumiendo que se ha alcanzado el problema actual y se  va a seguir la alternativa <code> a </code>.
060         * La alternativa puede ser descartada si <code> getObjetivoEstimado(a) &gt; =  AlgoritmoPD.getMejorValor() </code>,
061         * puesto que el valor objetivo de la solución obtnida estaría por encima del mejor valor obtenido hasta el momento.
062         * Es decir nos quedamos con las alternativas que cumplen 
063         * <code> getObjetivoEstimado(a) &lt;   AlgoritmoPD.getMejorValor() </code>.
064         * Donde <code> AlgoritmoPD.getMejorValor() </code> es el mejor valor encontrado por el algoritmo hasta ahora.
065         * <li> Si el problema original fuera de maximización el valor debe ser 
066         * una cota superior del valor de la propiedad objetivo del problema inicial, 
067         * asumiendo que se ha alcanzado el problema actual y se  va a seguir la alternativa <code> a </code>.
068         *  Es decir nos quedamos con las alternativas que cumplen 
069         * <code> getObjetivoEstimado(a) &gt;   AlgoritmoPD.getMejorValor() </code>.
070         * </ul>
071         * <p> Si no es posible estimar esa cota entonces devolverá MIN_VALUE si el problema es de minimización
072         *  y MAX_VALUE si es de maximización</p>
073         * @param a - Alternativa elegida
074         * @return El valor objetivo estimado si seguimos <code> a </code>
075         */
076        Double getObjetivoEstimado(A a);
077        
078}
