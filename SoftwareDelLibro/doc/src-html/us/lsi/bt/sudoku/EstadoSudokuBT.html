001package us.lsi.bt.sudoku;
002
003import java.util.Collections;
004import java.util.List;
005import java.util.stream.Collectors;
006
007import com.google.common.collect.Lists;
008
009import us.lsi.bt.EstadoBT;
010import us.lsi.sa.sudoku.CuadroSudoku;
011
012public class EstadoSudokuBT implements EstadoBT<CuadroSudoku, Integer> {
013
014        public static EstadoSudokuBT create(List<Integer> alternativasYaEscogidas) {
015                return new EstadoSudokuBT(alternativasYaEscogidas);
016        }
017
018        public static EstadoSudokuBT create() {
019                return new EstadoSudokuBT(Lists.newArrayList());
020        }
021        
022        private int pos;
023        private List<Integer> alternativasYaEscogidas;
024        private CuadroSudoku cuadro;
025                
026        private EstadoSudokuBT(List<Integer> alternativasYaEscogidas) {
027                super();
028                this.pos = alternativasYaEscogidas.size();
029                this.alternativasYaEscogidas = alternativasYaEscogidas;
030                this.cuadro = CuadroSudoku.create(completa(this.alternativasYaEscogidas));
031        }
032        
033        private List<Integer> completa(List<Integer> ls){
034                List<Integer> plus = Collections.nCopies(CuadroSudoku.numPosicionesLibres-ls.size(), 0);
035                List<Integer> r = Lists.newArrayList(ls);
036                r.addAll(plus);
037                return r;
038        }
039        
040        @Override
041        public Tipo getTipo() {
042                return Tipo.AlgunasSoluciones;
043        }
044
045        @Override
046        public EstadoBT<CuadroSudoku, Integer> getEstadoInicial() {
047                return EstadoSudokuBT.create();
048        }
049
050        
051        @Override
052        public EstadoSudokuBT avanza(Integer a) {
053                this.alternativasYaEscogidas.add(a);
054                this.pos = this.alternativasYaEscogidas.size(); 
055                this.cuadro = CuadroSudoku.create(completa(this.alternativasYaEscogidas));
056                return this;
057        }
058
059        @Override
060        public EstadoSudokuBT retrocede(Integer a) {
061                this.alternativasYaEscogidas.remove(pos-1);
062                this.pos = this.alternativasYaEscogidas.size();
063                this.cuadro = CuadroSudoku.create(completa(this.alternativasYaEscogidas));
064                return this;
065        }
066
067        @Override
068        public int size() {
069                return CuadroSudoku.numPosicionesLibres-this.pos;
070        }
071
072        @Override
073        public boolean esCasoBase() {
074                boolean r = this.pos == CuadroSudoku.numPosicionesLibres;
075                return r;
076        }
077
078        @Override
079        public List<Integer> getAlternativas() {
080                List<Integer> ls = cuadro.getValoresLibresEnPos(CuadroSudoku.posicionesLibres.get(this.pos)).stream().collect(Collectors.toList());
081                return ls;
082        }
083
084        @Override
085        public CuadroSudoku getSolucion() {
086                return this.cuadro;
087        }       
088
089        @Override
090        public Double getObjetivoEstimado(Integer a) {
091                return Double.MAX_VALUE;
092        }
093
094        public List<Integer> getAlternativasYaEscogidas() {
095                return alternativasYaEscogidas;
096        }
097
098        
099        
100}
