053        private  ListaDeAnunciosAEmitir(List<Integer> anunciosDecididosParaEmitir){
054                this.anunciosDecididosParaEmitir = Lists.newArrayList(anunciosDecididosParaEmitir);
055                this.anunciosDecididosParaEmitirSet = Sets.newHashSet(anunciosDecididosParaEmitir);
056                calculaPropiedadesDerivadas();          
057                calculaAnunciosDisponibles();
058        }

064        public boolean cumpleRestricciones() {
065                return this.numAnunciosIncompatibles == 0 && this.numAnunciosRepetidos == 0 && this.tiempoRestante >=0 ;                
066        }
067
068        private void calculaPropiedadesDerivadas(){                     
069                this.tiempoConsumido = 0;
070                this.valor = 0.;
071                for(int i=0; i< anunciosDecididosParaEmitir.size();i++){
072                        Integer a = anunciosDecididosParaEmitir.get(i);
073                        this.valor = this.valor+ProblemaAnuncios.getAnuncio(a).getPrecio(this.tiempoConsumido+1);
074                        this.tiempoConsumido=this.tiempoConsumido+ProblemaAnuncios.getAnuncio(a).getDuracion();
075                }
076                this.tiempoRestante = ProblemaAnuncios.tiempoTotal-this.tiempoConsumido;        
077                this.numAnunciosIncompatibles = 0;                              
078                for(ParInteger p: ProblemaAnuncios.restricciones){
079                        if(this.anunciosDecididosParaEmitirSet.contains(p.p1) && this.anunciosDecididosParaEmitirSet.contains(p.p2)){
080                                this.numAnunciosIncompatibles = this.numAnunciosIncompatibles +1;
081                        }
082                }
083                this.numAnunciosRepetidos =  this.anunciosDecididosParaEmitir.size() - this.anunciosDecididosParaEmitirSet.size();      
084        }
085        
086        
087        private void calculaAnunciosDisponibles(){              
088                Set<Integer> disponibles = Sets.newHashSet(ProblemaAnuncios.todosLosAnuncios);  
089                disponibles.removeAll(this.anunciosDecididosParaEmitirSet);
090                for(ParInteger p: ProblemaAnuncios.restricciones){
091                        if(this.anunciosDecididosParaEmitirSet.contains(p.p1)){
092                                disponibles.remove(p.p2);
093                        }
094                }
095                Set<Integer> quitar = Sets.newHashSet();
096                for(Integer e : disponibles){
097                        if(ProblemaAnuncios.getAnuncio(e).getDuracion()>this.tiempoRestante){
098                                quitar.add(e);
099                        }
100                }
101                disponibles.removeAll(quitar);
102                Comparator<Integer> cmp = Comparator.<Integer,Anuncio>comparing(x->ProblemaAnuncios.getAnuncio(x), Comparator.<Anuncio>reverseOrder());
103                this.anunciosDisponibles = Sets.newTreeSet(cmp);
104                this.anunciosDisponibles.addAll(disponibles);
105        }

107        public ListaDeAnunciosAEmitir insertar(int pos, Integer e){
108                Preconditions.checkPositionIndex(pos, this.anunciosDecididosParaEmitir.size());
109                Preconditions.checkArgument(!this.anunciosDecididosParaEmitirSet.contains(e));
110                List<Integer> ls = Lists.newArrayList(this.anunciosDecididosParaEmitir);
111                ls.add(pos, e);
112                return create(ls);
113        }
114        
115        public ListaDeAnunciosAEmitir insertarUltimo(Integer e){
116                return insertar(this.anunciosDecididosParaEmitir.size(),e);
117        }
118        
119        public ListaDeAnunciosAEmitir eliminar(int pos){
120                Preconditions.checkElementIndex(pos, this.anunciosDecididosParaEmitir.size());
121                List<Integer> ls = Lists.newArrayList(this.anunciosDecididosParaEmitir);
122                ls.remove(pos);
123                return create(ls);
124        }
125        
126        public ListaDeAnunciosAEmitir eliminarUltimo(){
127                Preconditions.checkArgument(!this.anunciosDecididosParaEmitir.isEmpty());
128                return eliminar(this.anunciosDecididosParaEmitir.size());
129        }
130
131        public ListaDeAnunciosAEmitir intercambiar(int i, int j){
132                Preconditions.checkElementIndex(i, this.anunciosDecididosParaEmitir.size());
133                Preconditions.checkElementIndex(j, this.anunciosDecididosParaEmitir.size());
134                Preconditions.checkArgument(i!=j);
135                List<Integer> ls = Lists.newArrayList(this.anunciosDecididosParaEmitir);
136                Lists2.intercambia(ls, i, j);
137                return create(ls);
138        }

184        public ParInteger getAlternativaInsertar() {
185                Preconditions.checkState(!this.anunciosDisponibles.isEmpty());
186                Integer pos = Math2.getEnteroAleatorio(0,this.anunciosDecididosParaEmitir.size() + 1);
187                List<Integer> ls = Lists.newArrayList(this.anunciosDisponibles);
188                Integer r = Math2.getEnteroAleatorio(0,ls.size());
189                return ParInteger.create(pos, ls.get(r));               
190        }
191
192        public ParInteger getAlternativaIntercambiar(){
193                return Math2.getParAleatorioYDistinto(0, this.anunciosDecididosParaEmitir.size());      
194        }

196        public List<AlternativaAnuncios.Opcion> getTiposDeOpcionesAlternativasPosibles(){
197                List<AlternativaAnuncios.Opcion> ls = Lists.newArrayList();
198                for(AlternativaAnuncios.Opcion op : AlternativaAnuncios.Opcion.values()){
199                        switch(op){
200                        case Insertar :
201                                if(!this.getAnunciosDisponibles().isEmpty()){
202                                        ls.add(op);
203                                }
204                                break;
205                        case Eliminar :
206                                if(!this.getAnunciosDecididosParaEmitir().isEmpty()){
207                                        ls.add(op);
208                                }
209                                break;
210                        case Intercambiar :
211                                if(this.getAnunciosDecididosParaEmitir().size() >=2){
212                                        ls.add(op);
213                                }
214                                break;
215                        }                       
216                }
217                return ls;
218        }

220        public static ListaDeAnunciosAEmitir getSolucionVoraz(){
221                ListaDeAnunciosAEmitir e = ListaDeAnunciosAEmitir.create();
222                while(!e.getAnunciosDisponibles().isEmpty()){
223                        Integer a = e.getAnunciosDisponibles().first();
224                        e = e.insertarUltimo(a);
225                }
226                return e;
227        }
