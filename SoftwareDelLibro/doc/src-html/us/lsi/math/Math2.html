048        private static MatrizFibonacci powMatrizFibonacci(Integer n){
049                MatrizFibonacci r;              
050                if(n==0){
051                        r = MatrizFibonacci.getOne();
052                }else if(n==1){
053                        r = MatrizFibonacci.getBase();
054                }else {
055                        r = powMatrizFibonacci(n/2);
056                        r = r.getSquare();
057                        if(n%2!=0){
058                                r = r.getMultiplyBase();
059                        }
060                }
061                return r;
062        }

065         * @pre n &gt; = 0
066         * @param n Un Entero
067         * @return El número n de Fibonacci calculado reduciendo el problema a una potenciación de matrices
068         * y en una versión iterativa
069         */
070        public static BigInteger fibonacci5(Integer n){
071                BigInteger ar,br;
072                BigInteger au,bu;
073                BigInteger at,bt;
074                BigInteger dos = new BigInteger("2");
075                ar = BigInteger.ONE;
076                br = BigInteger.ZERO;
077                au = BigInteger.ZERO;
078                bu = BigInteger.ONE;
079                while(n>0){
080                        if(n%2==1){
081                                at = bu.multiply(ar).add(au.multiply(ar)).add(au.multiply(br));
082                                bt = au.multiply(ar).add(bu.multiply(br));
083                                au = at;
084                                bu = bt;
085                        }
086                        at = ar.multiply(ar).add(dos.multiply(ar).multiply(br));
087                        bt = ar.multiply(ar).add(br.multiply(br));
088                        ar = at;
089                        br = bt;
090                        n =n/2;
091                }
092                return au;
093        }

096         * @pre n &gt; = 0
097         * @param n Un Entero
098         * @return El número n de Fibonacci calculado reduciendo el problema a una potenciación de matrices
099         * y en una versión iterativa y usando Long en vez de BigInteger
100         */
101        public static Long fibonacci3(Integer n){
102                Long ar,br;
103                Long au,bu;
104                Long at,bt;
105                ar = 1L;
106                br = 0L;
107                au = 0L;
108                bu = 1L;
109                while(n>0){
110                        if(n%2==1){
111                                at = bu*ar+au*ar+au*br;
112                                bt = au*ar+bu*br;
113                                au = at;
114                                bu = bt;
115                        }
116                        at = ar*ar+2*ar*br;
117                        bt = ar*ar+br*br;
118                        ar = at;
119                        br = bt;
120                        n =n/2;
121                }
122                return au;
123        }

127         * @pre base &gt; 0
128         * @param base Base de la potencia
129         * @param n Exponente de la potencia
130         * @return base &#94; n en una versión iterativa de complejidad log(n)
131         */
132        public static Double powi(Double base, Integer n){
133                Double r, u;
134                r = base;
135                u = 1.;
136                while( n > 0){
137               if(n%2==1){
138                             u = u * r;
139                        }
140                        r = r * r;
141                        n = n/2;
142                }
143                return u;
144
145        }

150         * @param n Exponente de la potencia
151         * @return base &#94; n en una versión iterativa de complejidad log(n)
152         */
153        public static Double powr(Double base, Integer n){
154                Double r;               
155                if(n==0){
156                        r = 1.;
157                }else if(n==1){
158                        r = base;
159                }else {
160                        r = powr(base,n/2);
161                        r = r*r;
162                        if(n%2!=0){
163                                r = r*base;
164                        }
165                }
166                return r;
167        }

181         * @param b Un segundo entero
182         * @return Calcula el máximo común divisor de los valores ablsolutos de los parámetros
183         */
184        public static int mcd(int a, int b){            
185               int u = Math.abs(a);
186               int v = Math.abs(b);
187               int r;
188               while (v != 0) {
189                    r = u % v;
190                    u = v;
191                    v = r;
192               }
193                    return u;
194        }

208        /**
209         * @pre b &gt; a
210         * @param a Límite inferior
211         * @param b Límte Superior
212         * @return Un entero aleatorio r tal que a &lt; = r &lt; b
213         */
214        public static Integer getEnteroAleatorio(Integer a, Integer b){         
215                Integer valor;
216                        Preconditions.checkArgument(b>a,a+","+b);
217                if(b-a == 1){
218                        valor = a;
219                }else{
220                        valor = a+rnd.nextInt(b-a);
221                }
222                return valor;
223        }

259        /**
260         * @param increment El incremento &delta;
261         * @param t Temperatura
262         * @return Si &delta; &lt; 0 devuelve 1, si t = 0 devuelve 0, en otro caso devuelve e &#94; (- &delta;/t) 
263         */
264        public static double boltzmann(double increment, double t){
265                double r;
266                if(increment < 0.){
267                        r = 1.;
268                } else if(t==0.){
269                        r = 0.;
270                } else {
271                        r = Math.exp(-increment/t);
272                }
273                return r;
274        }

278         * @param t Temperatura

282                        double rd = Math2.getDoubleAleatorio(0., 1.);

289         * @param probabilities Es una distribución de probabilidades para una variables aleatoria 
290         * con valores 0 hasta probabilidades.size() no incluido.
291         * @return Un entero entre 0 y probabilidades.size(), no incluido, con las probababilidades proporcionadas
292         */
293        public static Integer escogeEntre(List<Double> probabilities){
294                Preconditions.checkArgument(!probabilities.isEmpty());
295                Double ppa = 0.;
296                Integer r = 0;
297                double na = Math2.getDoubleAleatorio(0., 1.);
298                for(Double p:probabilities){
299                        ppa = ppa+p;
300                        if(ppa >= na){
301                                break;
302                        }
303                        r++;
304                }
305                return r;
306        }
