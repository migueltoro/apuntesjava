022public class OtrosEjemplos {
023
024        private static <T> Consumer<T> imprimeEnConsola() {
025                return x -> System.out.println(x);
026        }
027
028        private static <T, R> BiConsumer<T, R> imprimeEnConsola2() {
029                return (x, y) -> System.out.println(x + "," + y);
030        }
031
032        private static IntConsumer imprimeEnConsolaInt() {
033                return x -> System.out.println(x);
034        }
035
036        public static void ejemplo1() {
037
038                Double r = DoubleStream.iterate(7.0, x -> x * x).filter(x -> x > 3000)
039                                .findFirst().getAsDouble();
040
041                System.out.println("r= " + r);
042
043        }
044
045        public static void ejemplo2() {
046                IntStream.range(23, 550).forEach(imprimeEnConsolaInt());
047        }
048
049        public static void ejemplo3() {
050                Stream.concat(IntStream.range(50, 60).boxed(),
051                                IntStream.range(2, 20).boxed()).forEach(imprimeEnConsola());
052        }
053
054        public static void ejemplo4() {
055
056                IntStream.range(0, 5).boxed().forEach(imprimeEnConsola());
057
058                System.out.println("========");
059
060                IntStream.range(10, 15).boxed().forEach(imprimeEnConsola());
061
062                System.out.println("========");
063
064                Stream2.create(IntStream.range(0, 5).boxed())
065                                .<Integer, Boolean, Integer> joint(
066                                                Stream2.create(IntStream.range(10, 15).boxed()),
067                                                x -> x % 2 == 0, x -> x % 3 == 0, (x, y) -> x + y)
068                                .forEach(imprimeEnConsola());
069
070        }
071
072        public static void ejemplo5() {
073                Stream2<Long> s0  = Stream2.of(1L, 2L, 3L, 4L, 5L);
074                Stream2<Long> s1 = Stream2.of(4L, 10L, 9L, 29L);
075                Stream2<Long> s2 = Stream2.of(5L, 15L, 20L, 39L);
076                
077                
078                s0.concat(s1)
079                                .joint(s2, x -> x % 2, x -> x % 3, (x, y) -> x + y)
080                                .groupBy2(x -> x % 10, x -> x % 2, Collectors.summarizingLong(x -> x))
081                                .forEach(imprimeEnConsola2());
082        }
083
084        public static void ejemplo6() {
085                Set<Integer> s1 = Sets.newHashSet(1, 3, 5);
086                Set<Integer> s2 = Sets.newHashSet(16, 13, 15);
087                Stream2.create(s1.stream())
088                                .cartesianProduct(s2.stream(),
089                                                (x, y) -> ParInteger.create(x, y))
090                                .forEach(imprimeEnConsola());
091        }
092
093        public static void ejemplo7() {
094                Stream2.iterate(3L, x -> x * x).whileIncluded(x -> 81L - x >= 0)
095                                .forEach(imprimeEnConsola());
096        }
097
098        public static void ejemplos8() {
099                Stream2.of(1L, 2L, 3L, 4L, 5L)
100                                .joint(Stream.of(4L, 10L, 9L, 29L), x -> x % 5, x -> x % 3, (x, y) -> x +  y)
101                                .forEach(imprimeEnConsola());
102        }
103
104        public static void ejemplos9() {
105                Stream2.<Integer>iterate(0, x -> x + 4)
106                                .whileIncluded(x -> x <= 1000)
107                                .toList().forEach(imprimeEnConsola());
108        }
