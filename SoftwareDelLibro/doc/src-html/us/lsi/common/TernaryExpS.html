001package us.lsi.common;
002
003public class TernaryExpS<R> extends TernaryExp<R, R, R, R> {
004
005        TernaryExpS(Exp<R> op1, Exp<R> op2, Exp<R> op3,
006                        us.lsi.common.TernaryExp.TriFunction<R, R, R, R> operator,
007                        String symbol) {
008                super(op1, op2, op3, operator, symbol);
009        }
010
011        TernaryExpS(
012                        us.lsi.common.TernaryExp.TriFunction<R, R, R, R> operator,
013                        String symbol) {
014                super(operator, symbol);
015        }
016
017        @Override
018        public TernaryExpS<R> asTernary() {
019                return this;
020        }
021        
022        @Override
023        public Exp<R> clone() {
024                Exp<R> e1 = this.op1;
025                Exp<R> e2 = this.op2;
026                Exp<R> e3 = this.op3;
027                if(e1!= null) e1 = e1.clone();
028                if(e2!= null) e2 = e2.clone();
029                if(e3!= null) e3 = e3.clone();
030                return Exp.createTernaryS(e1, e2, e3, this.operator, this.symbol);
031        }
032}
