001package us.lsi.common;
002
003import com.google.common.base.Preconditions;
004
005import java.util.List;
006import java.util.function.BiFunction;
007import java.util.function.BiPredicate;
008import java.util.function.Function;
009import java.util.function.Predicate;
010
011
012
013public class Par<T1, T2> {
014
015        public static <T1, T2> Par<T1, T2> create(T1 p1, T2 p2) {
016                return new Par<T1, T2>(p1, p2);
017        }
018
019        public static <T1, T2> Par<T1, T2> create(Par<T1, T2> t) {
020                return new Par<T1, T2>(t.getP1(), t.getP2());
021        }
022        
023        public static <T> Par<T, T> create(List<T> t) {
024                Preconditions.checkArgument(t.size()==2);
025                return create(t.get(0), t.get(1));
026        }
027        
028        public static <T> Par<T, T> create(T[] t) {
029                Preconditions.checkArgument(t.length==2);
030                return create(t[0], t[1]);
031        }
032        
033        public  static <A,R,S,E,U> Par<A,Par<S,U>>  transformP2N2(Par<A,Par<S,E>> tt, Function<E,U> f){ 
034                return create(tt.getP1(),create(tt.getP2().getP1(),f.apply(tt.getP2().getP2())));
035        }
036        
037        public static <E1,E2> Predicate<Par<E1,E2>> predicate(BiPredicate<E1,E2> p){
038                return (Par<E1,E2> pp)->p.test(pp.getP1(), pp.getP2());
039        }
040        
041        public static <E1,E2,R> Function<Par<E1,E2>,R> function(BiFunction<E1,E2,R> f){
042                return (Par<E1,E2> pp)->f.apply(pp.getP1(), pp.getP2());
043        }
044        
045        public T1 p1;
046        public T2 p2;
047        protected Par(T1 p1, T2 p2) {
048                super();
049                this.p1 = p1;
050                this.p2 = p2;
051        }
052        public T1 getP1() {
053                return p1;
054        }
055        public T2 getP2() {
056                return p2;
057        }
058        
059        public Par<T1,T2> setP1(T1 v){
060                return create(v,getP2());
061        }
062        
063        public Par<T1,T2> setP2(T2 v){
064                return create(getP1(),v);
065        }
066        
067        public <R> Par<R,T2> transformP1(Function<T1,R> f){
068                return create(f.apply(getP1()),getP2());
069        }
070        
071        public <R> Par<T1,R> transformP2(Function<T2,R> f){
072                return create(getP1(),f.apply(getP2()));
073        }
074                
075        public boolean isP1(Predicate<T1> p){
076                return p.test(getP1());
077        }
078        
079        public boolean isP2(Predicate<T2> p){
080                return p.test(getP2());
081        }
082        
083        public boolean is(BiPredicate<T1,T2> p){
084                return p.test(getP1(),getP2());
085        }       
086        
087        
088        
089        @Override
090        public String toString() {
091                return "(" + p1 + "," + p2 + ")";
092        }
093        
094        @Override
095        public int hashCode() {
096                final int prime = 31;
097                int result = 1;
098                result = prime * result + ((p1 == null) ? 0 : p1.hashCode());
099                result = prime * result + ((p2 == null) ? 0 : p2.hashCode());
100                return result;
101        }
102        
103        @Override
104        public boolean equals(Object obj) {
105                if (this == obj)
106                        return true;
107                if (obj == null)
108                        return false;
109                if (getClass() != obj.getClass())
110                        return false;
111                Par<?,?> other = (Par<?, ?>) obj;
112                if (p1 == null) {
113                        if (other.p1 != null)
114                                return false;
115                } else if (!p1.equals(other.p1))
116                        return false;
117                if (p2 == null) {
118                        if (other.p2 != null)
119                                return false;
120                } else if (!p2.equals(other.p2))
121                        return false;
122                return true;
123        }
124        
125        
126        
127}
