001package us.lsi.common;
002
003public class Preconditions {
004
005        /**
006         * Checks that the boolean is true. Use for validating arguments to methods.
007         * @param condition A condition
008         */
009        public static void checkArgument(boolean condition){
010                if(!condition){
011                        throw new IllegalArgumentException();
012                }
013        }
014        
015        /**
016         * Checks that the boolean is true. Use for validating arguments to methods.
017         * @param message A message
018         * @param condition A condition
019         */
020        public static void checkArgument(boolean condition, String message){
021                if(!condition){
022                        throw new IllegalArgumentException(message);
023                }
024        }
025
026        /**
027         * Checks some state of the object, not dependent on the method arguments.  
028         * @param condition A condition
029         */
030        public static void checkState(boolean condition){
031                if(!condition){
032                        throw new IllegalArgumentException();
033                }
034        }
035        /**
036         * Checks some state of the object, not dependent on the method arguments. 
037         * @param message 
038         * @param condition A condition
039         */
040        public static void checkState(boolean condition, String message){
041                if(!condition){
042                        throw new IllegalArgumentException(message);
043                }
044        }
045        
046        /**
047         * Checks that the value is not null. 
048         * Returns the value directly, so you can use checkNotNull(value) inline.
049         * @return reference
050         */
051        public static <T> T checkNotNull(T reference){
052                if(reference == null){
053                        throw new NullPointerException(String.format("Es nulo %s", reference));
054                }
055                return reference;
056        }
057                
058        /**
059         * Checks that index is a valid element index into a list, string, or array with the specified size. 
060         * An element index may range from 0 inclusive to size exclusive. 
061         * You don't pass the list, string, or array directly; you just pass its size. 
062         * @param index Un índice 
063         * @param size 
064         * @return Index.
065         */
066        public static int checkElementIndex(int index, int size){
067                if(!(index>=0 && index<size)){
068                        throw new IndexOutOfBoundsException(String.format("Index = %d, size %d", index,size));
069                }
070                return index;
071        }
072        
073        /**
074         * Checks that index is a valid position index into a list, string, or array with the specified size. 
075         * A position index may range from 0 inclusive to size inclusive. 
076         * You don't pass the list, string, or array directly; you just pass its size. Returns index.
077         * @param index
078         * @param size
079         * @return Index.
080         */
081        public static int checkPositionIndex(int index, int size){
082                if(!(index>=0 && index<=size)){
083                        throw new IndexOutOfBoundsException(String.format("Index = %d, size %d", index,size));
084                }
085                return index;
086        }
087        
088        
089}
