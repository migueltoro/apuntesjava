001package us.lsi.concurrent.filosofos;
002
003import org.jdesktop.beansbinding.Converter;
004
005public class ConvertTextToLong extends Converter<String, Long> {
006
007        public ConvertTextToLong() {
008                // TODO Auto-generated constructor stub
009        }
010
011        @Override
012        public Long convertForward(String s) {
013                // TODO Auto-generated method stub
014                return new Long(s);
015        }
016
017        @Override
018        public String convertReverse(Long e) {
019                // TODO Auto-generated method stub
020                return e.toString();
021        }
022
023}
