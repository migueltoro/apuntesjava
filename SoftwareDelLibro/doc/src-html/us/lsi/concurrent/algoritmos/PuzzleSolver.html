001package us.lsi.concurrent.algoritmos;
002import java.util.concurrent.atomic.*;
003
004public class PuzzleSolver<P, M> extends ConcurrentPuzzleSolver<P, M> {
005        PuzzleSolver(Puzzle<P, M> puzzle) {
006                super(puzzle);
007        }
008
009        private final AtomicInteger taskCount = new AtomicInteger(0);
010
011        protected Runnable newTask(P p, M m, PuzzleNode<P, M> n) {
012                return new CountingSolverTask(p, m, n);
013        }
014
015        class CountingSolverTask extends SolverTask {
016                CountingSolverTask(P pos, M move, PuzzleNode<P, M> prev) {
017                        super(pos, move, prev);
018                        taskCount.incrementAndGet();
019                }
020
021                public void run() {
022                        try {
023                                super.run();
024                        } finally {
025                                if (taskCount.decrementAndGet() == 0)
026                                        solution.setValue(null);
027                        }
028                }
029        }
030
031}
