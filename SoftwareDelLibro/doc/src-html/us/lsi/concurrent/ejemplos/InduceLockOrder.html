001package us.lsi.concurrent.ejemplos;
002
003public class InduceLockOrder {
004
005        private static final Object tieLock = new Object();
006
007        public void transferMoney(final Account fromAcct, final Account toAcct,
008                        final DollarAmount amount) throws InsufficientFundsException {
009                
010                int fromHash = System.identityHashCode(fromAcct);
011                int toHash = System.identityHashCode(toAcct);
012
013                if (fromHash < toHash) {
014                        synchronized (fromAcct) {
015                                synchronized (toAcct) {
016                                        new Helper().transfer(fromAcct,toAcct,amount);
017                                }
018                        }
019                } else if (fromHash > toHash) {
020                        synchronized (toAcct) {
021                                synchronized (fromAcct) {
022                                        new Helper().transfer(fromAcct,toAcct,amount);
023                                }
024                        }
025                } else {
026                        synchronized (tieLock) {
027                                synchronized (fromAcct) {
028                                        synchronized (toAcct) {
029                                                new Helper().transfer(fromAcct,toAcct,amount);
030                                        }
031                                }
032                        }
033                }
034        }
035
036        interface DollarAmount extends Comparable<DollarAmount> {
037        }
038
039        interface Account {
040                void debit(DollarAmount d);
041
042                void credit(DollarAmount d);
043
044                DollarAmount getBalance();
045
046                int getAcctNo();
047        }
048
049        class InsufficientFundsException extends Exception {
050
051                /**
052                 * 
053                 */
054                private static final long serialVersionUID = 1L;
055        }
056
057        class Helper {
058                
059                public void transfer(final Account fromAcct, final Account toAcct,
060                                final DollarAmount amount) throws InsufficientFundsException {
061                        
062                        if (fromAcct.getBalance().compareTo(amount) < 0)
063                                throw new InsufficientFundsException();
064                        else {
065                                fromAcct.debit(amount);
066                                toAcct.credit(amount);
067                        }
068                }
069
070        }
071        
072}
