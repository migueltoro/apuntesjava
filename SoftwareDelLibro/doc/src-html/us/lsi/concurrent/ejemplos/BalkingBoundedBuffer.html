001package us.lsi.concurrent.ejemplos;
002
003
004import net.jcip.annotations.*;
005
006@ThreadSafe
007public class BalkingBoundedBuffer<V> extends BaseBoundedBuffer<V> {
008
009        
010        public BalkingBoundedBuffer(int capacity) {
011                super(capacity);
012        }
013
014        @GuardedBy("this")
015        public synchronized void put(V v) throws BufferFullException {
016                if (isFull())
017                        throw new BufferFullException();
018                doPut(v);
019        }
020        @GuardedBy("this")
021        public synchronized V take()throws BufferEmptyException {
022                if (isFull())
023                        throw new BufferEmptyException();
024                return doTake();
025        }
026}
