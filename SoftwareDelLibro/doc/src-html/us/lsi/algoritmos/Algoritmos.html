128        /**
129         * @param fichero Fichero que  describe laq red de flujo
130         * @return Algoritmo de red de flujo que resuelve el problema especificado en el fichero
131         */
132        public static FlowAlgorithm createFlow(String fichero) {
133                FlowGraph f = new FlowGraph(FlowEdge.factory);
134                f = (FlowGraph) GraphsReader.newGraph(fichero, FlowVertex.factory, FlowEdge.factory,f);
135                return FlowAlgorithm.create(f);
136        }

138        /**
139         * Un algoritmo AStar para ir del vértice de inicio hasta el vértice destino
140         * 
141         * @param <V> Tipo del vértice

146         * @return Algortimo AStar
147         * 
148         */
149        public static <V, E> AStarAlgorithm<V, E> createAStar(
150                        AStarGraph<V, E> graph, V startVertex, V endVertex) {

152        }

154         * Un algoritmo AStar para ir del vértice de inicio hasta el  primer vértice que cumple el predicado

161         */

163        public static <V, E> AStarAlgorithm<V, E> createAStar(
