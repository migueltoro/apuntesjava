001package us.lsi.pd.tareasprocesadores;
002
003import java.util.List;
004import java.util.stream.Collectors;
005
006import us.lsi.stream.Stream2;
007
008public class Tarea {
009
010        public static List<Tarea> tareas;
011        
012        public static void leeTareas(String fichero){
013                tareas = Stream2.fromFile(fichero)
014                                .map(s ->Tarea.create(s))
015                                .collect(Collectors.toList());
016        }
017        
018        public static Tarea create(String s) {
019                return new Tarea(s);
020        }
021
022        public static Tarea create(Integer id, Double duracion) {
023                return new Tarea(id, duracion);
024        }
025
026        private Double duracion;
027        private Integer id;
028        private Tarea(Integer id, Double duracion) {
029                super();
030                this.duracion = duracion;
031                this.id = id;
032        }
033        private Tarea(String s) {
034                super();
035                String[] elementos = s.split(",");              
036                this.id = new Integer(elementos[0].trim());
037                this.duracion = new Double(elementos[1].trim());;
038        }
039        public Double getDuracion() {
040                return duracion;
041        }
042        public Integer getId() {
043                return id;
044        }
045        @Override
046        public int hashCode() {
047                final int prime = 31;
048                int result = 1;
049                result = prime * result
050                                + ((duracion == null) ? 0 : duracion.hashCode());
051                result = prime * result + ((id == null) ? 0 : id.hashCode());
052                return result;
053        }
054        @Override
055        public boolean equals(Object obj) {
056                if (this == obj)
057                        return true;
058                if (obj == null)
059                        return false;
060                if (!(obj instanceof Tarea))
061                        return false;
062                Tarea other = (Tarea) obj;
063                if (duracion == null) {
064                        if (other.duracion != null)
065                                return false;
066                } else if (!duracion.equals(other.duracion))
067                        return false;
068                if (id == null) {
069                        if (other.id != null)
070                                return false;
071                } else if (!id.equals(other.id))
072                        return false;
073                return true;
074        }
075        @Override
076        public String toString() {
077                return "Tarea [id= " + id + ", duracion= " + duracion + "]";
078        }
079        
080        
081
082}
