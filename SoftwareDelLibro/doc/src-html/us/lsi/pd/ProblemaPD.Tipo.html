001package us.lsi.pd;
002import java.util.*;
003
004import us.lsi.pd.AlgoritmoPD.Sp;
005
006
007/**
008 * <p>Interface que debe implementar un problema para ser resuelto por la técnica de Programación Dinámica </p>
009 * 
010 * <p>La documentación puede encontarse en el: <a href="../../../document/Tema14.pdf" target="_blank">Tema14</a></p>
011 * 
012 * @author Miguel Toro
013 *
014 * @param <S> El tipo de la solución del problema
015 * @param <A> El tipo de las alternativas
016 * @see AlgoritmoPD.Sp
017 */
018public interface ProblemaPD<S,A>  {     
019        
020        public enum Tipo{Min,Max,Todas}
021        
022        
023        /**
024         * @return El tipo del problema: minimiza, maximiza o busca un valor que involucra a todas las alternativas
025         *
026         * 
027         **/
028        Tipo getTipo(); 
029        /**
030         * @return El tamaño del problema
031         */
032        int size();
033        /**
034         * @return Si el problema es un caso base
035         */
036        boolean esCasoBase();
037        /**
038         * @pos 
039         * <ul>
040         *      <li> Si el problema no tiene solución se debe devolver null
041         *      <li> Si el problema tiene solución asociada a una alternativa esta será de la forma (a,p).Siendo a la alternativa y p la propiedad correspondiente.
042         *      <li> Si el problema tiene solución pero no asociada a una alternativa esta será de la forma (null,p).Siendo p la propiedad correspondiente.
043         *      </ul> 
044         * @return  La solución del caso base
045         */
046        Sp<A> getSolucionCasoBase();
047        /**
048         * @pre ls!=null, ls no está vacía, ls no contiene null, ls no contiene valores de la forma <code> (a,null) </code>
049         * 
050         * @pos 
051         * <ul>
052         *  <li> Si ls es vacía tras eliminar los valores <code> (a,null) </code> la solución devuelta es null. El problema no tiene solución.
053         *      <li> Si el problema tiene solución asociada a una alternativa, esta será de la forma <code>(a,p)</code>. Siendo <code> a </code>la alternativa y <code> p </code> la propiedad correspondiente. 
054         *      <li> Si el problema tiene solución pero no asociada a una alternativa esta será de la forma <code> (null,p)</code>. Siendo <code> p </code> la propiedad correspondiente.
055         *      </ul> 
056         * @param ls - Soluciones existentes alcanzados tras tomar las diferentes alternativas
057         * @return La solución del problema
058         */
059        Sp<A> seleccionaAlternativa(List<Sp<A>> ls);
060        /**
061         * @pre 0 &lt; = i &lt; getNumeroSubProblemas(a)
062         * @param a - Alternativa escogida
063         * @param np - Número del subproblema
064         * @return Subproblema
065         */
066        ProblemaPD<S,A> getSubProblema(A a, int np);
067        /**
068         * @pre <code> ls.size() == getNumeroSubProblemas(a) </code>, a está incluida en <code> getAlternativas() </code> y
069         * ls no contiene null
070         * @param a - Alternativa escogida
071         * @param ls - Soluciones de los subproblemas
072         * @return La solución del problema alcanzado siguiendo la alternativa <code> a </code> 
073         * y las soluciones parciales de los subproblemas del mismo están en ls 
074         */
075        Sp<A> combinaSolucionesParciales(A a , List<Sp<A>> ls); 
076        /**
077         * @pos Si un problema no tiene solución el conjunto de alternativas es vacío 
078         * @return Las alternativas disponibles para el problema
079         */
080        List<A> getAlternativas();
081        /**
082         * @param a - Alternativa escogida
083         * @return Número de subproblemas para una alternativa dada
084         */
085        int getNumeroSubProblemas(A a);
086        /**
087         * @pre El problema es un caso base
088         * @pos 
089         * <ul>
090         *  <li> Si sp es de la forma <code> (a,p) </code> la solución se reconstruye a partir de esos dos valores. 
091         *  <li> Si es de la forma <code> (null,p) </code> sólo a partir de <code> p </code>.
092         * </ul> 
093         * @param sp - Solución parcial del caso base
094         * @return La solución para un caso base si la solución parcial es sp. 
095         */
096        S getSolucionReconstruida(Sp<A> sp);    
097        /**
098         * @pre 
099         * <p> El problema no es un caso base, <code> ls !=null </code>, <code> ls.size() &gt; 0 </code> y ls no contiene null.</p>
100         * <p> Si el tipo es Min o Max entonces sp = (a,p) </p>
101         * <p> Si tipo es Todas entonces sp puede ser (a,p) o (null,p). Entonces  ls tiene en cada caso el significado correspondiente </p>
102         * @pos
103         * <ul>
104         *  <li> Si <code> sp = (a,p) </code> la solución se reconstruye a partir de <code> a, p, ls </code> estando <code> ls </code> las 
105         *  de los subproblemas del problema que se alcanza siguiendo la alternativa <code> a </code> 
106         *  <li> Si <code> sp = (null,p) </code> la solución se reconstruye a partir de las 
107         *  soluciones de los problemas que tienen solución y se alcanzan siguiendo las diferentes alternativas posibles.
108         * </ul> 
109         * @param sp - Solución parcial del problema
110         * @param ls - 
111         * <ul>
112         *  <li> Si <code> sp = (a,p)  </code> la solución de los subproblemas del problema que se alcanza 
113         *  siguiendo la alternativa <code> a </code>. 
114         *  <li> Si <code> sp = (null,p) </code>  las soluciones de los problemas que tienen solución y 
115         *  se alcanzan siguiendo las diferentes alternativas posibles. 
116         * </ul>
117         * Soluciones obtenidas siguiendo la diferentes alternativas
118         * @return La solución para el problema. 
119         */
120        S getSolucionReconstruida(Sp<A> sp, List<S> ls);        
121        
122        /**
123         * @pos 
124         * <ul>
125         * <li> Si el problema es de minimización el valor debe ser 
126         * una cota inferior del valor de la propiedad objetivo del problema inicial, 
127         * asumiendo que se ha alcanzado el problema actual y se  va a seguir la alternativa <code> a </code>.
128         * La alternativa puede ser descartada si <code> getObjetivoEstimado(a) &gt; =  AlgoritmoPD.getMejorValor() </code>,
129         * puesto que el valor objetivo de la solución obtnida estaría por encima del mejor valor obtenido hasta el momento.
130         * Es decir nos quedamos con las alternativas que cumplen 
131         * <code> getObjetivoEstimado(a) &lt;   AlgoritmoPD.getMejorValor() </code>.
132         * Donde <code> AlgoritmoPD.getMejorValor() </code> es el mejor valor encontrado por el algoritmo hasta ahora.
133         * <li> Si el problema original fuera de maximización el valor debe ser 
134         * una cota superior del valor de la propiedad objetivo del problema inicial, 
135         * asumiendo que se ha alcanzado el problema actual y se  va a seguir la alternativa <code> a </code>.
136         *  Es decir nos quedamos con las alternativas que cumplen 
137         * <code> getObjetivoEstimado(a) &gt;   AlgoritmoPD.getMejorValor() </code>.
138         * </ul>
139         * <p> Si no es posible estimar esa cota entonces devolverá MIN_VALUE si el problema es de minimización
140         *  y MAX_VALUE si es de maximización</p>
141         * @param a - Alternativa elegida
142         * @return El valor objetivo estimado si seguimos <code> a </code>
143         */
144        Double getObjetivoEstimado(A a);
145        /**
146         * Se asume que se llama al método sólo cuando se ha resuelto el problema y por lo tanto sabemos el valor de la propiedad solución.
147         * En muchos caso se puede calcular sumando un valor acumlado al valor conocido de la solución.
148         * @pos Si no es posible calcularla  se devolverá MAX_VALUE si el problema es de minimización y MIN_VALUE si es de maximización. 
149         * @return Valor de propiedad objetivo del problema inicial asumiendo que estamos en el problema actual   
150         */
151        Double getObjetivo();
152}
