001package us.lsi.pd.secuencias;
002
003import java.util.ArrayList;
004import java.util.List;
005
006import us.lsi.common.Tuple2;
007import us.lsi.pd.secuencias.Secuencia.Accion;
008
009public class SolucionSecuencias {
010        
011        
012        public static SolucionSecuencias create() {
013                return new SolucionSecuencias();
014        }
015
016        List<Tuple2<Accion,Integer>> acciones;
017        
018        private SolucionSecuencias() {
019                super();
020                this.acciones = new ArrayList<>();
021        }
022
023        public List<Tuple2<Accion, Integer>> getAcciones() {
024                return acciones;
025        }
026
027        public void addFirst(Tuple2<Accion,Integer> p) {
028                acciones.add(0,p);
029        }
030        
031        @Override
032        public String toString() {
033                return "SolucionSecuencias [acciones=" + acciones + "]";
034        }
035        
036        
037}
