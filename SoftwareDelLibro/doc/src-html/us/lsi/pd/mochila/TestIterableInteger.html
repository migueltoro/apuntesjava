004import java.util.stream.Collectors;
005import java.util.stream.IntStream;
006import java.util.*;
007
008import com.google.common.collect.Lists;
009
010public class TestIterableInteger {
011
012        
013        public static void main(String[] args) {
014                // TODO Auto-generated method stub
015                System.out.println(getAlternativas());
016                System.out.println(getAlternativas());
017                System.out.println(Lists.newArrayList(getAlternativas()));
018                System.out.println(getAlternativas());
019                for(Integer e: getAlternativas()){
020                        System.out.println("     "+e);
021                }
022        }
023        
024        public static Iterable<Integer> getAlternativas() {
025                return IntStream.rangeClosed(0, 27)
026                        .boxed()
027                        .filter(x-> x%2==0)
028                        .sorted(Comparator.<Integer>reverseOrder())
029                        .collect(Collectors.<Integer>toList());
030        }
031
