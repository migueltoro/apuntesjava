001package us.lsi.pd;
002
003import java.util.List;
004
005import com.google.common.base.Preconditions;
006
007import us.lsi.pd.AlgoritmoPD.Sp;
008
009/**
010 * <p> Una adaptación del {@link us.lsi.pd.ProblemaPD ProblemaPD} para el caso de reducción.
011 * </p>
012 * 
013 * <p>Una explicación puede encontrarse en <a href="../../../document/ExpPD.html" target="_blank">Programación Dinámica</a></p>
014 * 
015 * 
016 * @author Miguel Toro
017 *
018 * @param <S> El tipo de la solución
019 * @param <A> El tipo de la alternativa
020 */
021
022public interface ProblemaPDR<S, A> extends ProblemaPD<S, A> {
023        
024        @Override
025        default int getNumeroSubProblemas(A a){
026                return 1;
027        }
028        
029        @Override
030        default ProblemaPD<S,A> getSubProblema(A a, int np) {
031                Preconditions.checkArgument(np==0);
032                return getSubProblema(a);
033        }
034        
035        ProblemaPD<S,A> getSubProblema(A a);
036        
037        @Override
038        default Sp<A> getSolucionParcialPorAlternativa(A a , List<Sp<A>> ls){
039                return getSolucionParcialPorAlternativa(a ,ls.get(0));
040        }
041        
042        Sp<A> getSolucionParcialPorAlternativa(A a ,Sp<A> sp);
043        
044        @Override
045        default S getSolucionReconstruidaCasoRecursivo(Sp<A> sp, List<S> ls){
046                return getSolucionReconstruidaCasoRecursivo(sp,ls.get(0));
047        }
048        
049        S getSolucionReconstruidaCasoRecursivo(Sp<A> sp, S ls);
050        
051        
052
053}
