001package us.lsi.astar.puzzle;
002
003import java.util.*;
004import java.util.stream.Collectors;
005
006
007
008
009import com.google.common.collect.Lists;
010
011import us.lsi.common.*;
012import us.lsi.graphs.SimpleEdge;
013import us.lsi.graphs.VirtualVertex;
014
015public class EstadoPuzzle implements VirtualVertex<EstadoPuzzle,SimpleEdge<EstadoPuzzle>> {
016        
017        /**
018         * @param d Lista de valores del puzzle dados por filas de abajo arriba
019         * @return Un EstadoPuzzle
020         */
021        public static EstadoPuzzle create(Integer... d) {
022                return new EstadoPuzzle(d);
023        }
024        
025        private Integer[][] datos; 
026        private int i0;
027        private int j0;
028        public static int numFilas = 3;
029        
030        private EstadoPuzzle(Integer... d) {
031                super(); 
032                this.datos = new Integer[EstadoPuzzle.numFilas][EstadoPuzzle.numFilas];
033                Set<Integer> s = new HashSet<Integer>();        
034                int i=0,j=0;
035                for (int e:d) {                                                 
036                                        if(e<0 || e> (EstadoPuzzle.numFilas*EstadoPuzzle.numFilas-1)){
037                                                throw new IllegalArgumentException();
038                                        }
039                                        if(e==0){
040                                                this.i0 =i;
041                                                this.j0 =j;
042                                        }
043                                        s.add(e);
044                                        this.datos[i][j] = e;
045                                        if(j== EstadoPuzzle.numFilas-1){
046                                                i++;
047                                                j=0;
048                                        }else{
049                                                j++;
050                                        }
051                                
052                }
053                if(d.length!= EstadoPuzzle.numFilas*EstadoPuzzle.numFilas || s.size()!= EstadoPuzzle.numFilas*EstadoPuzzle.numFilas)
054                        throw new IllegalArgumentException();
055        }
056
057
058        public EstadoPuzzle(Integer[][] datos, int i0, int j0) {
059                super();
060                this.datos = datos;
061                this.i0 = i0;
062                this.j0 = j0;
063        }
064
065        public Integer getDato(int i, int j) {
066                if(i<0 || i>=EstadoPuzzle.numFilas || j<0 || j>=EstadoPuzzle.numFilas)
067                        throw new IllegalArgumentException();
068                return datos[i][j];
069        }
070        
071        public EstadoPuzzle getVecino(int f, int c){
072                if(f<0 || f>=EstadoPuzzle.numFilas || c<0 || c>=EstadoPuzzle.numFilas)
073                        throw new IllegalArgumentException();
074                Integer[][] dd = new Integer[EstadoPuzzle.numFilas][EstadoPuzzle.numFilas];
075                for (int i = 0; i < EstadoPuzzle.numFilas; i++) {
076                        for (int j = 0; j < EstadoPuzzle.numFilas; j++) {
077                                dd[i][j] = datos[i][j];
078                        }
079                }
080                dd[f][c] = datos[this.i0][this.j0];
081                dd[this.i0][j0] = datos[f][c];
082                return new EstadoPuzzle(dd,f,c);
083        }
084        
085        @Override
086        public Set<EstadoPuzzle> getNeighborListOf() {
087                List<ParInteger> ls = Lists.newArrayList(ParInteger.create(1,0),ParInteger.create(0,1),ParInteger.create(-1,0),ParInteger.create(0,-1));
088                return ls.stream()
089                                 .<ParInteger> map((ParInteger x)-> ParInteger.create(x.p1+this.i0, x.p2+this.j0))
090                                 .filter((ParInteger x)-> x.p1>=0 && x.p1<EstadoPuzzle.numFilas && x.p2>=0 && x.p2<EstadoPuzzle.numFilas)
091                                 .<EstadoPuzzle>map((ParInteger x)-> this.getVecino(x.p1,x.p2))
092                                 .collect(Collectors.<EstadoPuzzle>toSet());
093        }
094        
095        public Set<SimpleEdge<EstadoPuzzle>> edgesOf(){
096                return this.getNeighborListOf().stream()
097                   .map(x->SimpleEdge.create(this,x))
098                   .collect(Collectors.<SimpleEdge<EstadoPuzzle>>toSet());
099        }
100        
101        @Override
102        public boolean isNeighbor(EstadoPuzzle e) {
103                return this.getNumDiferentes(e)==2 && Math.abs(this.i0-e.i0)+Math.abs(this.j0-e.j0)==1;
104        }
105
106        public boolean isValid() {
107                Set<Integer> s = new HashSet<Integer>();        
108                for (int j = 0; j < EstadoPuzzle.numFilas; j++) {
109                        for (int i = 0; i < EstadoPuzzle.numFilas; i++) {
110                                s.add(datos[i][j]);
111                        }
112                }
113                return s.size()== EstadoPuzzle.numFilas*EstadoPuzzle.numFilas;
114        }
115        
116        public Integer getNumDiferentes(EstadoPuzzle e){
117                Integer s = 0;
118                for (int i = 0; i < EstadoPuzzle.numFilas; i++) {
119                        for (int j = 0; j < EstadoPuzzle.numFilas; j++) {
120                                if (!this.datos[i][j].equals(e.datos[i][j])) {
121                                        s++;
122                                }
123                        }
124                }
125                return s;
126        }
127        
128        @Override
129        public int hashCode() {
130                final int prime = 31;
131                int result = 1;
132                result = prime * result + Arrays.deepHashCode(datos);
133                return result;
134        }
135
136
137        @Override
138        public boolean equals(Object obj) {
139                if (this == obj)
140                        return true;
141                if (obj == null)
142                        return false;
143                if (!(obj instanceof EstadoPuzzle))
144                        return false;
145                EstadoPuzzle other = (EstadoPuzzle) obj;
146                if (!Arrays.deepEquals(datos, other.datos))
147                        return false;
148                return true;
149        }
150
151
152        @Override
153        public String toString() {
154                String s ="\n";
155                
156                for (int i = datos.length-1; i >= 0; i--) {
157                        for (int j = 0; j < datos.length; j++) {
158                                if (j<datos.length-1) {
159                                        s = s + datos[i][j]+",";
160                                }else {
161                                        s = s + datos[i][j]+"\n";
162                                }
163                        }
164                }
165                return s;
166        }
167        
168        
169}
