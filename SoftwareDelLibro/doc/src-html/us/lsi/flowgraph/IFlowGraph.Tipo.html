001package us.lsi.flowgraph;
002
003import org.jgrapht.DirectedGraph;
004
005/**
006 * Una implementación de un grafo dirigido adecuado para representar 
007 * una Red de Flujo. Es decir un grafo simple dirigido acíciclo
008 * 
009 * @author Miguel Toro
010 *
011 * @param <V> Tipo del vértice
012 * @param <E> Tipo de las aristas
013 */
014
015public interface IFlowGraph<V, E> extends DirectedGraph<V, E> {
016        
017        /**
018         * Un vértice de una red de Flujo puede ser una Fuente, un Sumidero o
019         * un vértice intermedio
020         *
021         */
022        public enum Tipo{Source,Sink,Intermedio};       
023        /* 
024         * @see org.jgrapht.Graph#getEdgeWeight(java.lang.Object)
025         */
026        double getEdgeWeight(E edge);
027        /**
028         * @param edge Un arista
029         * @return La cota mínima del flujo que puede pasar por la arista
030         */
031        double getMinEdgeWeight(E edge);
032        /**
033         * @param edge Una arista
034         * @return El coste unitario de transportar algo por la arista
035         */
036        double getUnitEdgeWeight(E edge);
037        /**
038         * @param vertex Un vértice
039         * @return Si el vértice es una fuente
040         */
041        boolean isSource(V vertex);     
042        /**
043         * @param vertex Un vértice 
044         * @return Si el vértice es un sumidero
045         */
046        boolean isSink(V vertex);       
047        /**
048         * @param vertex Un vértice
049         * @return La cota máxima del flujo que pasa por el vértice. Es decir de la suma del flujo
050         * que entra menos el que sale
051         */
052        double getMaxVertexWeight(V vertex);
053        /**
054         * @param vertex Un vértice
055         * @return La cota mínima del flujo que pasa por el vértice. Es decir de la suma del flujo
056         * que entra menos el que sale
057         */
058        double getMinVertexWeight(V vertex);
059        /**
060         * @param vertex Un vértice
061         * @return El coste unitario del flujo que pasa por el vértice. Es decir de la suma del flujo
062         * que entra menos el que sale
063         */
064        double getUnitVertexWeight(V vertex);           
065}
