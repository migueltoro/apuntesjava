001package us.lsi.flowgraph;
002import org.jgraph.graph.DefaultEdge;
003
004/**
005 * Una arista simple de una Red de Fujo.
006 * Cada arista de este tipo tiene asociado un coste unitario, un flujo máximo y otro mínimo 
007 * 
008 * @author Miguel Toro
009 *
010 */
011public class FlowEdge extends DefaultEdge {
012        /**
013         * 
014         */
015        private static final long serialVersionUID = 1L;
016        
017        private Double min=0.;
018        private Double max=Double.MAX_VALUE;
019        private Double cost=0.;
020        
021        public static FlowEdge createEdge(FlowVertex v1, FlowVertex v2, String[] formato) {
022                return new FlowEdge(v1,v2,formato);
023        }
024        
025        public static FlowEdge createEdge(FlowVertex v1, FlowVertex v2) {
026                return new FlowEdge(v1,v2);
027        }
028        
029        private FlowEdge(FlowVertex from, FlowVertex to, Double min, Double max,Double cost)  {
030                super();
031                super.source=from;
032                super.target =to;
033                this.min = min;
034                this.max = max;
035                this.cost = cost;
036        }
037        
038        private FlowEdge(FlowVertex from, FlowVertex to) {
039                super();
040                super.source=from;
041                super.target =to;
042                this.min = 0.;
043                this.max = Double.MAX_VALUE;
044                this.cost = 0.;
045        }
046        
047        private FlowEdge(FlowVertex from, FlowVertex to,String[] formato) {
048                super();
049                super.source=from;
050                super.target =to;
051                this.min = formato[2].equals("inf")?Double.MAX_VALUE:new Double(formato[2]);
052                this.max = formato[3].equals("inf")?Double.MAX_VALUE:new Double(formato[3]);
053                this.cost = formato[4].equals("inf")?Double.MAX_VALUE:new Double(formato[4]);
054        }
055        
056        public Double getMin() {
057                return min;
058        }
059
060        public Double getMax() {
061                return max;
062        }
063
064        public Double getCost() {
065                return cost;
066        }
067
068        
069        public FlowVertex getFrom() {
070                return (FlowVertex) super.source;
071        }
072
073        public FlowVertex getTo() {
074                return (FlowVertex) super.target;
075        }
076
077        
078        
079        @Override
080        public String toString() {
081                return source.toString() + "--" + target.toString();
082        }       
083        
084        @Override
085        public int hashCode() {
086                final int prime = 31;
087                int result = 1;
088                result = prime * result + ((source == null) ? 0 : source.hashCode());
089                result = prime * result + ((target == null) ? 0 : target.hashCode());
090                return result;
091        }
092
093        @Override
094        public boolean equals(Object obj) {
095                if (this == obj)
096                        return true;
097                if (obj == null)
098                        return false;
099                if (!(obj instanceof FlowEdge))
100                        return false;
101                FlowEdge other = (FlowEdge) obj;
102                if (source == null) {
103                        if (other.source != null)
104                                return false;
105                } else if (!source.equals(other.source))
106                        return false;
107                if (target == null) {
108                        if (other.target != null)
109                                return false;
110                } else if (!target.equals(other.target))
111                        return false;
112                return true;
113        }
114
115        
116
117}
