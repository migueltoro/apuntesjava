020/**
021 * 
022 * Resuelve le problema del  cálculo de los caminos con aristas disjuntas mediante una red de flujo
023 * 
024 * @author Miguel Toro
025 *
026 */
027public class CaminosAristasDisjuntas {
028        
029        public static void creaFichero(String fileIn, String fileOut, Ciudad from, Ciudad to){
030                Graph<Ciudad,Carretera> g =  new SimpleWeightedGraph<Ciudad,Carretera>(Carretera::create);
031                g =  GraphsReader.newGraph(fileIn,Ciudad::create, Carretera::create,g,Carretera::getKm);
032                SimpleDirectedGraph<Ciudad,Carretera> gs = Graphs2.toDirectedGraph((SimpleGraph<Ciudad,Carretera>) g);
033                PrintWriter f = null;
034                try {
035                        f = new PrintWriter(new BufferedWriter(new FileWriter(fileOut)));
036                } catch (IOException e) {
037                        System.err.println(e.toString());
038                }
039                
040                f.println("#VERTEX#");
041                for(Ciudad c:gs.vertexSet()){
042                        if (c.equals(from)) {
043                                f.println(c + ",1,0.0,inf,1.0");
044                        }else if(c.equals(to)){
045                                f.println(c + ",2,0.0,inf,0.0");
046                        }else {
047                                f.println(c + ",0,0.0,inf,0.0"); //No hay limitaciones para que pasen caminos
048                                                                 //por cada vértice
049                        }
050                }               
051                f.println("#EDGE#");
052                for(Carretera cr:gs.edgeSet()){
053                        Ciudad source = gs.getEdgeSource(cr);
054                        Ciudad target = gs.getEdgeTarget(cr);
055                        if (!source.equals(to) && !target.equals(from)) {
056                                f.println(source + "," + target+ ",0.0,1.0,0.0");
057                        }
058                }
059                f.close();
060        }
061        
062        public static void main(String[] args) {
063                Ciudad from = Ciudad.create("Cadiz");
064                Ciudad to = Ciudad.create("Almeria");
065                CaminosAristasDisjuntas.creaFichero("./ficheros/andalucia.txt","./ficheros/andaluciaFlow2.txt",from,to);
066                FlowGraphSolution fs = SolveFlowGraphProblem.solve(
067                                "./ficheros/andaluciaFlow2.txt", 
068                                "./ficheros/andaluciaFlowGrafo2.gv", 
069                                "./ficheros/andaluciaFlowSoluciones2.gv",
070                                "./ficheros/andaluciaFlowConstraints2.txt",
071                                true,
072                                false);
073
074                System.out.println(fs.getOptFlow());
075                System.out.println("====");
076                System.out.println("=========="+fs.getWalks(false));
077                System.out.println("=========="+fs.getWalks(true));
078        }
079}
