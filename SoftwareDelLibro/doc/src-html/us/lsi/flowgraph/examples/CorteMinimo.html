001package us.lsi.flowgraph.examples;
002
003
004import java.util.Set;
005import org.jgrapht.UndirectedGraph;
006import org.jgrapht.alg.StoerWagnerMinimumCut;
007import org.jgrapht.alg.flow.EdmondsKarpMFImpl;
008import org.jgrapht.graph.SimpleWeightedGraph;
009
010import us.lsi.graphs.GraphsReader;
011import us.lsi.graphs.examples.Carretera;
012import us.lsi.graphs.examples.Ciudad;
013
014/**
015 * Resuelve varios problemas de corte mínimo
016 * 
017 * @author Miguel Toro
018 *
019 */
020public class CorteMinimo {
021
022        public static UndirectedGraph<Ciudad,Carretera> creaFichero(String fileIn){
023                
024                UndirectedGraph<Ciudad,Carretera> g =  new SimpleWeightedGraph<Ciudad,Carretera>(Carretera::create);
025                g =  (UndirectedGraph<Ciudad,Carretera>) GraphsReader.newGraph(fileIn,Ciudad::create, Carretera::create,g,Carretera::getKm);
026                return g;
027        }
028        
029        
030        public static void main(String[] args) {
031                
032                UndirectedGraph<Ciudad,Carretera> graph = CorteMinimo.creaFichero("./ficheros/andalucia.txt");
033                StoerWagnerMinimumCut<Ciudad,Carretera> a = new StoerWagnerMinimumCut<Ciudad,Carretera>(graph);
034                Set<Ciudad> source = a.minCut();
035                System.out.println(source);
036                EdmondsKarpMFImpl<Ciudad,Carretera> a2 = new EdmondsKarpMFImpl<Ciudad,Carretera>(graph);
037                Ciudad from = Ciudad.create("Antequera");
038                Ciudad to = Ciudad.create("Almeria");
039                Double r = a2.calculateMinCut(from, to);
040                Set<Ciudad> source2 = a2.getSourcePartition();
041                Set<Ciudad> target2 = a2.getSinkPartition();
042                Set<Carretera> edges = a2.getCutEdges();
043                System.out.println(r+","+source2+","+target2+","+edges);                
044        }
045
046}
