001package us.lsi.graphs.examples;
002
003
004import org.jgrapht.graph.DefaultWeightedEdge;
005
006public class Carretera extends DefaultWeightedEdge {
007
008        /**
009         * 
010         */
011        private static final long serialVersionUID = 1L;
012
013        public static Carretera create(Ciudad c1, Ciudad c2) {
014                return new Carretera(c1,c2);
015        }
016
017        public static Carretera create(Ciudad c1, Ciudad c2, String[] formato) {
018                return new Carretera(c1,c2,formato);
019        }
020
021        private static int num =0;
022        private Ciudad source;
023        private Ciudad target;
024        private String nombre;
025        private double km;
026        private int id;
027
028        public Carretera(Ciudad c1, Ciudad c2) {
029                super();
030                this.source = c1;
031                this.target = c2;
032                this.nombre = "";
033                this.km = 0.;
034                this.id = num;
035                num++;
036        }
037        
038        public Carretera(Ciudad c1, Ciudad c2, String[] nombre) {
039                super();
040                this.source = c1;
041                this.target = c2;
042                this.nombre = nombre[2];
043                this.km = new Double(nombre[3]);
044                this.id = num;
045                num++;
046        }
047
048        public String getNombre() {
049                return nombre;
050        }
051
052
053        public double getKm() {
054                return km;
055        }
056
057        
058        public Ciudad getSource(){
059                return source;
060        }
061        
062        public Ciudad getTarget(){
063                return target;
064        }
065
066        @Override
067        public String toString() {
068                return "("+super.getSource()+","+super.getTarget()+","+this.nombre+","+super.getWeight()+")";
069        }
070
071        @Override
072        public int hashCode() {
073                final int prime = 31;
074                int result = 1;
075                result = prime * result + id;
076                return result;
077        }
078
079        @Override
080        public boolean equals(Object obj) {
081                if (this == obj)
082                        return true;
083                if (obj == null)
084                        return false;
085                if (!(obj instanceof Carretera))
086                        return false;
087                Carretera other = (Carretera) obj;
088                if (id != other.id)
089                        return false;
090                return true;
091        }
092
093        
094
095        
096}
