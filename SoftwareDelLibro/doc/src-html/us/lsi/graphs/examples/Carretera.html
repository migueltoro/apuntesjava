001package us.lsi.graphs.examples;
002
003import org.jgraph.graph.DefaultEdge;
004
005import us.lsi.graphs.StringEdgeFactory;
006
007public class Carretera extends DefaultEdge {
008
009        /**
010         * 
011         */
012        private static final long serialVersionUID = 1L;
013
014        public static StringEdgeFactory<Ciudad, Carretera> factoria = new Factoria();
015
016        public Carretera(Ciudad c1, Ciudad c2) {
017                super();
018                super.source = c1;
019                super.target = c2;
020        }
021
022        
023        
024        @Override
025        public String toString() {
026                return "[]";
027        }
028
029
030
031        private static class Factoria implements
032                        StringEdgeFactory<Ciudad, Carretera> {
033
034                public Factoria() {
035                        super();
036                        // TODO Auto-generated constructor stub
037                }
038
039                @Override
040                public Carretera createEdge(Ciudad c1, Ciudad c2) {
041                        // TODO Auto-generated method stub
042                        return new Carretera(c1, c2);
043                }
044
045                @Override
046                public Carretera createEdge(Ciudad c1, Ciudad c2, String[] formato) {
047                        // TODO Auto-generated method stub
048                        return createEdge(c1, c2);
049                }
050        }
051}
