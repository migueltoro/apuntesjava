001package us.lsi.graphs;
002
003import java.util.List;
004import java.util.Map;
005
006import org.jgrapht.Graph;
007
008import com.google.common.base.Preconditions;
009import com.google.common.collect.Lists;
010import com.google.common.collect.Maps;
011
012/**
013 * @author Miguel Toro 
014 *
015 * @param <V> Tipo de los vértices
016 * @param <E> Tipo de las aristas
017 * 
018 * <a> Una vista de un grafo dónde cada vértice se identifica con un entero </a>
019 */
020public class GraphView<V, E> {
021
022        /**
023         * @param grafo Un grafo
024         * @return la vista del grafo
025         * @param <V> El tipo de los vértices
026         * @param <E> El tipo de las aristas
027         */
028        public static <V, E> GraphView<V, E> create(Graph<V, E> grafo) {
029                return new GraphView<V, E>(grafo);
030        }
031
032        private List<V> vertices;
033        private Graph<V,E> grafo;
034        private Integer numVertices;
035        private Map<V,Integer> map;
036        
037        private GraphView(Graph<V,E> grafo) {
038                this.grafo = grafo;
039                this.vertices = Lists.newArrayList(grafo.vertexSet());
040                this.map = Maps.newHashMap();
041                Integer n = 0;
042                for(V v:vertices){
043                        map.put(v,n);
044                        n++;
045                }
046                this.numVertices = this.vertices.size();
047        }
048
049        /**
050         * @return La lista de vértices del grafo
051         */
052        public List<V> getVertices() {
053                return vertices;
054        }
055
056        /**
057         * @return El grafo original
058         */
059        public Graph<V, E> getGrafo() {
060                return grafo;
061        }
062
063        /**
064         * @return El número de vértices
065         */
066        public Integer getNumVertices() {
067                return numVertices;
068        }
069
070        /**
071         * @pre i en el rango  0..getNumVertices()-1
072         * @param i Un índice en el rango 0..getNumVertices()-1
073         * @return El vértices de índice <code> i </code>
074         */
075        public V getVertice(int i) {
076                return vertices.get(i);
077        }
078        
079        /**
080         * @pre El grafo contiene al vértice v
081         * @param v Un vértice del grafo
082         * @return El entero correspondiente al vértice v
083         */
084        public Integer getIndex(V v) {
085                Preconditions.checkArgument(this.grafo.containsVertex(v));
086                return map.get(v);
087        }
088        
089        /**
090         * @pre hay una arista de i a j
091         * @param i Un índice válido
092         * @param j Un índice válido
093         * @return El peso de la arista que va de i a j
094         */
095        public Double getWeight(int i, int j) {
096                Preconditions.checkArgument(this.isEdge(i,j));
097                V v1 = getVertice(i);
098                V v2 = getVertice(j);
099                E cr = grafo.getEdge(v1,v2);
100                return grafo.getEdgeWeight(cr);
101        }
102        
103        /**
104         * @pre i, j en el rango  0..getNumVertices()-1
105         * @param i Un índice válido de un vértice
106         * @param j Un índice válido de un vértice
107         * @return Si hay una arista entre los dos vértices dados por sus identifadores
108         */
109        public boolean isEdge(int i, int j) {
110                V v1 = getVertice(i);
111                V v2 = getVertice(j);
112                E e = grafo.getEdge(v1,v2);
113                return e==null?false:true;
114        }
115        
116        /**
117         * @pre isEdge(i, j)
118         * @param i Un índice válido de un vértice
119         * @param j Un índice válido de un vértice
120         * @return La arista entre i y j
121         */
122        public E getEdge(int i, int j) {
123                Preconditions.checkArgument(this.isEdge(i,j));
124                V v1 = getVertice(i);
125                V v2 = getVertice(j);
126                E e = grafo.getEdge(v1,v2);
127                return e;
128        }
129}
