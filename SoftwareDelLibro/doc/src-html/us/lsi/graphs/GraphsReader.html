001package us.lsi.graphs;
002
003import java.io.File;
004import java.io.FileNotFoundException;
005import java.util.ArrayList;
006import java.util.HashMap;
007import java.util.List;
008import java.util.Map;
009import java.util.Scanner;
010
011import org.jgrapht.Graph;
012
013/**
014 * <p>
015 * Clase adecuada para construir un grafo a partir de la información en un fichero
016 * 
017 * El fichero debe tener una estructura como la siguiente. 
018 * 
019 * Cada línea en la sección de vértices representa un vértice. 
020 * El primer campo debe ser un identificador único para el mismo
021 * Los otros campos propiedades
022 * 
023 * Cada línea de la sección de aristas representa una arista
024 * Los dos primeros campos son los identificadores de los vértices extremos
025 * Los otros campos propiedades
026 * </p>
027 * 
028 * #VERTEX# <br>
029 * s1,Sevilla <br>
030 * s2,Córdoba <br>
031 * s3,Cádiz <br>
032 * s4,Málaga <br>
033 * #EDGE# <br>
034 * s1,s2,180. <br>
035 * s1,s3,120. <br>
036 * s2,s4,140. <br>
037 * s4,s1,210. <br>
038 *
039 * 
040 * 
041 * @author Miguel Toro
042 *
043 */
044public class GraphsReader {
045
046        private static String[] eliminaBlancos(String[] s) {
047                String[] r = new String[s.length];
048                for (int i = 0; i < r.length; i++) {
049                        r[i] = s[i].trim();
050                }
051                return r;
052        }
053
054        /**
055         * @param file Fichero con la información de entrada
056         * @param vf Factoría para construir los vértices a partir de las líneas del fichero previamente tokenizada
057         * @param ef Factoría para construir las aristas a partir de las líneas del fichero previamente tokenizada
058         * @param graph Grafo de entrada 
059         * @return Grafo con la información del fichero
060         * @param <V> El tipo de los vértices
061         * @param <E> El tipo de las aristas
062         * @throws IllegalArgumentException Si no encuentra el fichero o si el fichero tiene una estructura no adecuada
063         */
064        public static <V, E> Graph<V, E> newGraph(String file,
065                        StringVertexFactory<V> vf, StringEdgeFactory<V, E> ef,
066                        Graph<V, E> graph) {
067
068                Map<String, V> idVertices = new HashMap<>();
069                Graph<V, E> ret = graph;
070                Scanner sc = null;
071                try {
072                        sc = new Scanner(new File(file));
073                } catch (FileNotFoundException e1) {
074                        throw new IllegalArgumentException("Fichero " + file
075                                        + " no encontrado");
076                }
077
078                List<String> filas = new ArrayList<>();
079
080                while (sc.hasNext()) {
081                        filas.add(sc.next());
082                }
083
084                sc.close();
085                try {
086                        List<String> vertices = filas.subList(1, filas.indexOf("#EDGE#"));
087                        List<String> aristas = filas.subList(filas.indexOf("#EDGE#") + 1,
088                                        filas.size());
089
090                        for (String verticeStr : vertices) {
091                                String[] vertice = eliminaBlancos(verticeStr.split(","));
092
093                                if (vertice.length == 0) {
094                                        throw new IllegalArgumentException(
095                                                        "El fichero de entrada de vertices no es correcto. Linea: "
096                                                                        + verticeStr);
097                                }
098                                V vertex = vf.createVertex(vertice);
099                                ret.addVertex(vertex);
100                                idVertices.put(vertice[0], vertex);
101                        }
102
103                        for (String aristaStr : aristas) {
104                                String[] arista = eliminaBlancos(aristaStr.split(","));
105
106                                if (arista.length < 2)
107                                        throw new IllegalArgumentException(
108                                                        "El número de vértices de la arista no es correcto");
109
110                                if (arista.length == 2) {
111                                        ret.addEdge(idVertices.get(arista[0]),
112                                                        idVertices.get(arista[1]));
113                                } else if (arista.length > 2) {
114                                        E edge = ef.createEdge(idVertices.get(arista[0]),
115                                                        idVertices.get(arista[1]), arista);
116                                        ret.addEdge(idVertices.get(arista[0]),
117                                                        idVertices.get(arista[1]), edge);
118                                }
119                        }
120
121                        return ret;
122
123                } catch (IndexOutOfBoundsException e) {
124                        throw new IllegalArgumentException(
125                                        "La composición del fichero no es correcta");
126                }
127        }
128}
