001package us.lsi.graphs;
002
003import java.util.*;
004
005import org.jgrapht.*;
006
007/**
008 * <p> Implementación de un grafo virtual simple 
009 * Asumimos que los vértices son subtipo de VirtualVertex &lt; V,E &gt;
010 * Asumimos que las aristas son subtipos de SimpleEdge &lt; V &gt; 
011 * Si se invoca una operación para modificar el grafo se disparará 
012 * la excepción UnsupportedOperationException
013 * </p>
014 * 
015 * <p> El grafo es no es modificable por lo que no están permitadas las operación de moficación </p>
016 * 
017 * @see us.lsi.graphs.VirtualVertex
018 * 
019 *
020 * 
021 * @constraint Asumimos que los objetos creados del tipo VirtualVertex 
022 * deben ser vértices válidos del grafo virtual
023 * 
024 * @author Miguel Toro
025 *
026 * @param <V> El tipo de los vértices
027 * @param <E> El tipo de las aristas
028 * 
029 */
030public class SimpleVirtualGraph<V extends VirtualVertex<V,E>, E extends SimpleEdge<V>>
031        implements Graph<V, E> {
032        
033        
034        private EdgeFactory<V,E> edgeFactory;
035        private Set<V> vertexSet;
036                
037        protected SimpleVirtualGraph(EdgeFactory<V, E> edgeFactory, V[] vertex) {
038                super();
039                this.edgeFactory = edgeFactory;
040                this.vertexSet = new HashSet<V>();
041                for(V v:vertex){
042                        vertexSet.add(v);
043                }
044        }
045
046        protected SimpleVirtualGraph(EdgeFactory<V, E> edgeFactory){
047                super();
048                this.edgeFactory = edgeFactory;
049                this.vertexSet = new HashSet<V>();
050        }
051        
052        @Override
053        public EdgeFactory<V, E> getEdgeFactory() {
054                return edgeFactory;
055        }
056
057        @Override
058        public boolean containsEdge(E e) {
059                return e.getSource().isNeighbor(e.getTarget());
060        }
061
062        @Override
063        public boolean containsEdge(V v1, V v2) {
064                return v1.isNeighbor(v2);
065        }
066        @Override
067        public boolean containsVertex(V v) {
068                return true;
069        }
070        @Override
071        public V getEdgeSource(E e) {
072                return e.getSource();
073        }
074
075        @Override
076        public V getEdgeTarget(E e) {
077                return e.getTarget();
078        }
079
080        @Override
081        public double getEdgeWeight(E e) {
082                return 1.;
083        }
084        
085        @Override
086        public E getEdge(V v1, V v2) {
087                E a = null;
088                if (v1.isNeighbor(v2)) {
089                        a = getEdgeFactory().createEdge(v1, v2);
090                }
091                return a;
092        }
093        
094        @Override
095        public Set<E> getAllEdges(V v1, V v2) {
096                Set<E> s = new HashSet<>();
097                if (v1.isNeighbor(v2))
098                        s.add(getEdgeFactory().createEdge(v1, v2));
099                return s;
100        }
101        
102        @Override
103        public Set<V> vertexSet(){
104                return vertexSet; 
105        }
106        
107        @Override
108        public Set<E> edgesOf(V v) {
109                return v.edgesOf();
110        }
111        
112        public Set<V> getNeighborListOf(V v) {
113                return v.getNeighborListOf();
114        }
115        
116        
117        /* 
118         * @see org.jgrapht.Graph#edgeSet()
119         * @exception UnsupportedOperationException Si se invoca
120         */
121        @Override
122        public Set<E> edgeSet() {
123                throw new UnsupportedOperationException();
124        }
125        
126        /* 
127         * @see org.jgrapht.Graph#addEdge(java.lang.Object, java.lang.Object)
128         * @exception UnsupportedOperationException Si se invoca
129         */
130        @Override
131        public E addEdge(V arg0, V arg1) {
132                throw new UnsupportedOperationException();
133        }
134
135        
136        /* 
137         * @see org.jgrapht.Graph#addEdge(java.lang.Object, java.lang.Object, java.lang.Object)
138         * @exception UnsupportedOperationException Si se invoca
139         */
140        @Override
141        public boolean addEdge(V arg0, V arg1, E arg2) {
142                throw new UnsupportedOperationException();
143        }
144
145
146        /* 
147         * @see org.jgrapht.Graph#addVertex(java.lang.Object)
148         * @exception UnsupportedOperationException Si se invoca
149         */
150        @Override
151        public boolean addVertex(V arg0) {
152                throw new UnsupportedOperationException();
153        }
154
155        /*
156         * @see org.jgrapht.Graph#removeAllEdges(java.util.Collection)
157         * @exception UnsupportedOperationException Si se invoca
158         */
159        @Override
160        public boolean removeAllEdges(Collection<? extends E> arg0) {
161                throw new UnsupportedOperationException();
162        }
163
164        @Override
165        public Set<E> removeAllEdges(V arg0, V arg1) {
166                throw new UnsupportedOperationException();
167        }
168
169        /* 
170         * @see org.jgrapht.Graph#removeAllVertices(java.util.Collection)
171         * @exception UnsupportedOperationException Si se invoca
172         */
173        @Override
174        public boolean removeAllVertices(Collection<? extends V> arg0) {
175                throw new UnsupportedOperationException();
176        }
177
178        /* 
179         * @see org.jgrapht.Graph#removeEdge(java.lang.Object)
180         * @exception UnsupportedOperationException Si se invoca
181         */
182        @Override
183        public boolean removeEdge(E arg0) {
184                throw new UnsupportedOperationException();
185        }
186
187        /* 
188         * @see org.jgrapht.Graph#removeEdge(java.lang.Object, java.lang.Object)
189         * @exception UnsupportedOperationException Si se invoca
190         */
191        @Override
192        public E removeEdge(V arg0, V arg1) {
193                throw new UnsupportedOperationException();
194        }
195
196        /* 
197         * @see org.jgrapht.Graph#removeVertex(java.lang.Object)
198         * @exception UnsupportedOperationException Si se invoca
199         */
200        @Override
201        public boolean removeVertex(V arg0) {
202                throw new UnsupportedOperationException();
203        }       
204
205}
