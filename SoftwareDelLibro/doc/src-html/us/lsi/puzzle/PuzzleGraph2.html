001package us.lsi.puzzle;
002
003import java.util.Set;
004import java.util.function.Predicate;
005
006import org.jgrapht.EdgeFactory;
007
008import us.lsi.graphs.SimpleEdge;
009
010public class PuzzleGraph2 extends PuzzleGraph1 {
011
012        public static PuzzleGraph2 create(
013                        EdgeFactory<EstadoPuzzle, SimpleEdge<EstadoPuzzle>> edgeFactory) {
014                return new PuzzleGraph2(edgeFactory);
015        }
016
017        private PuzzleGraph2(EdgeFactory<EstadoPuzzle, SimpleEdge<EstadoPuzzle>> edgeFactory) {
018                super(edgeFactory);
019        }
020
021        @Override
022        public double getWeightToEnd(EstadoPuzzle startVertex,EstadoPuzzle endVertex, Predicate<EstadoPuzzle> goal,
023                        Set<EstadoPuzzle> goalSet) {
024                if(startVertex==null || goal==null)
025                        throw new IllegalArgumentException("El vértice inicial y el goal no pueden ser null");
026                return goal.test(startVertex)?0:1;
027        }
028}
