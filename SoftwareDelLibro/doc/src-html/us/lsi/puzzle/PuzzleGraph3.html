001package us.lsi.puzzle;
002
003import java.util.Set;
004import java.util.function.Predicate;
005
006import org.jgrapht.EdgeFactory;
007
008import us.lsi.graphs.SimpleEdge;
009
010public class PuzzleGraph3 extends PuzzleGraph1 {
011
012        public static PuzzleGraph3 create(
013                        EdgeFactory<EstadoPuzzle, SimpleEdge<EstadoPuzzle>> edgeFactory) {
014                return new PuzzleGraph3(edgeFactory);
015        }
016
017        private PuzzleGraph3(
018                        EdgeFactory<EstadoPuzzle, SimpleEdge<EstadoPuzzle>> edgeFactory) {
019                super(edgeFactory);
020        }
021
022        @Override
023        public double getWeightToEnd(EstadoPuzzle startVertex,EstadoPuzzle endVertex, Predicate<EstadoPuzzle> goal,
024                        Set<EstadoPuzzle> goalSet) {
025                if(startVertex==null || goalSet==null)
026                        throw new IllegalArgumentException("El vértice inicial y el goalSet no pueden ser null");
027                Double r = 0.;
028                for(EstadoPuzzle e: goalSet){
029                        r = r+startVertex.getNumDiferentes(e);
030                }       
031                return r;
032        }
033}
